\pagestyle{empty}

\newcommand{\platzkarte}[1]{%
\begin{tikzpicture}
    % Rahmen (Schnittlinie)
    \draw[fachfarbe, line width=1.5pt] (0,0) rectangle (45mm,88mm);

    % Platznummer (groß und fett)
    \node[anchor=north, fachfarbe, font=\fontsize{28}{32}\selectfont\bfseries]
        at (22.5mm,84mm) {P#1};

    % Namenslinie
    \draw[fachfarbe, line width=0.8pt] (5mm,68mm) -- (40mm,68mm);
    \node[anchor=north, font=\scriptsize, text=gray] at (22.5mm,67mm) {Name};

    % Punkte-Feld
    \draw[fachfarbe, line width=0.5pt] (5mm,45mm) rectangle (40mm,58mm);
    \node[anchor=north, font=\small\bfseries, fachfarbe] at (22.5mm,57mm) {Punkte};
    \node[anchor=center, font=\large] at (22.5mm,50mm) {\rule{15mm}{0.4pt} / \maxpunkte};

    % NP-Feld (Notenpunkte) oder Note
    \draw[fachfarbe, line width=0.5pt] (5mm,28mm) rectangle (40mm,41mm);
    \node[anchor=north, font=\small\bfseries, fachfarbe] at (22.5mm,40mm) {Note (NP)};
    \node[anchor=center, font=\large] at (22.5mm,33mm) {\rule{15mm}{0.4pt}};

    % QR-Code zur Stundenleistung (18mm Höhe)
    \node[anchor=south] at (22.5mm,2mm) {\qrcode[height=18mm]{\slurl}};
\end{tikzpicture}%
}
