% --- BEISPIELRECHNUNG ---
\abschnitt{Musterbeispiel}

\begin{beispielbox}
\textbf{Aufgabe:} Ein Auto fährt mit 72 km/h. Welche Strecke legt es in 5 Sekunden zurück?

\vspace{0.5em}
\textbf{Lösung:}
\begin{enumerate}[leftmargin=2em, itemsep=2pt]
    \item \textbf{Gegeben:} $v = 72$ km/h $= 20$ m/s, $t = 5$ s
    \item \textbf{Gesucht:} $s$
    \item \textbf{Formel:} $s = v \cdot t$
    \item \textbf{Einsetzen:} $s = 20\ein{m/s} \cdot 5\ein{s} = 100\ein{m}$
    \item \textbf{Antwort:} Das Auto legt 100 m zurück.
\end{enumerate}
\end{beispielbox}
