\abschnitt{Die zwei Algorithmen (Kernstruktur)}

\begin{hinweisbox}
\textbf{Algorithmus 1: Verhältnisformel aufstellen} (3 Schritte)
\begin{enumerate}[leftmargin=1.5em, itemsep=0pt]
    \item Elementsymbole aufschreiben (z.B. Li, O)
    \item Wertigkeiten ablesen (z.B. I, II)
    \item Kreuzregel: Wertigkeit A $\rightarrow$ Index B $\rightarrow$ Li\textsubscript{2}O
\end{enumerate}

\vspace{6pt}

\textbf{Algorithmus 2: Reaktionsgleichung aufstellen und ausgleichen} (4 Schritte)
\begin{enumerate}[leftmargin=1.5em, itemsep=0pt]
    \item Wortgleichung (Lithium + Sauerstoff $\rightarrow$ Lithiumoxid)
    \item Formelgleichung mit O\textsubscript{2} (Li + O\textsubscript{2} $\rightarrow$ Li\textsubscript{2}O)
    \item Atome zählen (Links/Rechts)
    \item Vorfaktoren ausgleichen (iterativ):\\
          Atom-Anzahlen unter Gleichung, dann:\\
          a) Produkt-Vorfaktor anpassen\\
          b) O\textsubscript{2}-Vorfaktor: O links = O rechts\\
          c) Metall-Vorfaktor: Metall links = rechts\\
          d) Kontrolle: Links = Rechts?
\end{enumerate}
\end{hinweisbox}

\abschnitt{Häufige Schülerfehler}

\begin{warnbox}
\textbf{Fehler 1: O statt O\textsubscript{2}}\\
SuS schreiben „Mg + O $\rightarrow$ MgO" statt „Mg + O\textsubscript{2} $\rightarrow$ MgO"\\
\textit{Reaktion:} Betonen, dass Sauerstoff in der Natur als O\textsubscript{2} vorliegt (Algorithmus 2, Schritt 2!).

\vspace{6pt}

\textbf{Fehler 2: Index statt Vorfaktor}\\
SuS schreiben „MgO\textsubscript{2}" statt „2 MgO"\\
\textit{Reaktion:} „Index ändert den STOFF! MgO\textsubscript{2} wäre ein völlig anderer Stoff. Vorfaktor ändert nur die Menge."

\vspace{6pt}

\textbf{Fehler 3: Kreuzregel nicht angewendet}\\
SuS schreiben „AlO" statt „Al\textsubscript{2}O\textsubscript{3}"\\
\textit{Reaktion:} Zurück zu Algorithmus 1: „Al hat Wertigkeit III, O hat II. Was ist das kgV von 2 und 3?"

\vspace{6pt}

\textbf{Fehler 4: Atome nicht richtig gezählt}\\
SuS zählen Moleküle statt Atome\\
\textit{Reaktion:} „In O\textsubscript{2} sind ZWEI O-Atome. In Fe\textsubscript{2}O\textsubscript{3} sind ZWEI Fe und DREI O."
\end{warnbox}

\abschnitt{Didaktische Hinweise}

\textbf{Warum „Händezahl" und nicht Elektronenkonfiguration?}
\begin{itemize}[leftmargin=1.5em, itemsep=2pt]
    \item Ionenbindung ist erst in Klasse 9 Thema (hier nur kurzer Hinweis!)
    \item Wertigkeiten werden pragmatisch als Tabellenwerte eingeführt
    \item Die Oktettregel erklärt nicht Fe(II/III) oder Cu(I/II)
    \item Metapher „Händezahl" ist für Kl. 8 anschaulich und ausreichend
\end{itemize}

\textbf{Warum zwei getrennte Algorithmen?}
\begin{itemize}[leftmargin=1.5em, itemsep=2pt]
    \item Klarere Trennung: Formel ermitteln (Alg. 1) vs. Gleichung ausgleichen (Alg. 2)
    \item Schüler erkennen, dass die Produktformel NICHT gegeben ist
    \item Jeder Algorithmus ist kurz und überschaubar (3 bzw. 4 Schritte)
    \item AB + LB enthalten beide Algorithmen zum Nachschlagen
\end{itemize}

\textbf{Ionenbindungs-Hinweis (neu):}
\begin{itemize}[leftmargin=1.5em, itemsep=2pt]
    \item Kurze Erklärung: Metall gibt Elektronen ab, O nimmt sie auf
    \item Verbindet Wertigkeit mit Elektronentransfer
    \item Bereitet Ionenbindung (Thema 3.3) vor, ohne zu weit zu gehen
    \item Aufgabe 1 fragt jetzt nach Elektronenzahl
\end{itemize}

\abschnitt{Lösungen Arbeitsblatt}

\textbf{Merksatz:}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item Die \textcolor{loesung}{Wertigkeit} gibt an, wie viele \textcolor{loesung}{Bindungen} ein Atom eingehen kann.
    \item Man kann sie sich als „\textcolor{loesung}{Händezahl}" vorstellen.
\end{itemize}

\textbf{Algorithmus 1 (Lücken):}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item Li\textsubscript{\textcolor{loesung}{2}}O\textsubscript{\textcolor{loesung}{1}} $\rightarrow$ Li\textsubscript{2}O
\end{itemize}

\textbf{Algorithmus 2 (Lücken):}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item Atome zählen: Links: Li = \textcolor{loesung}{1}, O = \textcolor{loesung}{2}; Rechts: Li = \textcolor{loesung}{2}, O = \textcolor{loesung}{1}
    \item Vorfaktoren ausgleichen (mit Atomzahlen unter Gleichung):
    \begin{itemize}[leftmargin=1em, itemsep=0pt]
        \item Start: Li=\textcolor{loesung}{1}, O=\textcolor{loesung}{2} | Li=\textcolor{loesung}{2}, O=\textcolor{loesung}{1} $\rightarrow$ nicht gleich!
        \item a) Produkt: 2 Li\textsubscript{2}O $\rightarrow$ Li=\textcolor{loesung}{1}/\textcolor{loesung}{2}, O=\textcolor{loesung}{4}/\textcolor{loesung}{2}
        \item b) Metall: 4 Li $\rightarrow$ Li=\textcolor{loesung}{4}/\textcolor{loesung}{2}, O=\textcolor{loesung}{4}/\textcolor{loesung}{2}
    \end{itemize}
    \item Ergebnis: \textcolor{loesung}{4} Li + \textcolor{loesung}{1} O\textsubscript{2} $\rightarrow$ \textcolor{loesung}{2} Li\textsubscript{2}O
\end{itemize}

\textbf{Aufgabe 1:} Mg = \textcolor{loesung}{II/2}, Al = \textcolor{loesung}{III/3}, O = \textcolor{loesung}{II/2}, Ca = \textcolor{loesung}{II/2}\\
(Elektronen: Mg gibt \textcolor{loesung}{2} ab, Al gibt \textcolor{loesung}{3} ab, O nimmt \textcolor{loesung}{2} auf, Ca gibt \textcolor{loesung}{2} ab)

\textbf{Aufgabe 2:}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item Ca(II), O(II) $\rightarrow$ \textcolor{loesung}{CaO}
    \item Zn(II), O(II) $\rightarrow$ \textcolor{loesung}{ZnO}
    \item Al(III), O(II) $\rightarrow$ \textcolor{loesung}{Al\textsubscript{2}O\textsubscript{3}}
    \item Fe(III), O(II) $\rightarrow$ \textcolor{loesung}{Fe\textsubscript{2}O\textsubscript{3}}
    \item Cu(II), O(II) $\rightarrow$ \textcolor{loesung}{CuO}
\end{itemize}

\textbf{Aufgabe 3:} (Geführte Beispiele + Selbstständig)
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item a) \textbf{Magnesium – einfacher Fall:}\\
          Start: Mg=1, O=2 | Mg=1, O=1 $\rightarrow$ nicht gleich!\\
          Produkt: 2 MgO $\rightarrow$ Mg=2, O=2\\
          Metall: 2 Mg $\rightarrow$ Mg=2, O=2\\
          Ergebnis: \textcolor{loesung}{2 Mg + 1 O\textsubscript{2} $\rightarrow$ 2 MgO}
    \item b) \textbf{Eisen – komplexer Fall:}\\
          Start: Fe=1, O=2 | Fe=2, O=3 $\rightarrow$ nicht gleich!\\
          Produkt: 2 Fe\textsubscript{2}O\textsubscript{3} $\rightarrow$ Fe=4, O=6\\
          O\textsubscript{2}: 3 O\textsubscript{2} $\rightarrow$ O=6\\
          Metall: 4 Fe $\rightarrow$ Fe=4\\
          Ergebnis: \textcolor{loesung}{4 Fe + 3 O\textsubscript{2} $\rightarrow$ 2 Fe\textsubscript{2}O\textsubscript{3}}
    \item c) \textbf{Aluminium – selbstständig:}\\
          Lösungsweg analog zu Eisen (gleiche Wertigkeit III)\\
          Ergebnis: \textcolor{loesung}{4 Al + 3 O\textsubscript{2} $\rightarrow$ 2 Al\textsubscript{2}O\textsubscript{3}}
    \item d) \textbf{Kupfer – selbstständig:}\\
          Lösungsweg analog zu Magnesium (gleiche Wertigkeit II)\\
          Ergebnis: \textcolor{loesung}{2 Cu + 1 O\textsubscript{2} $\rightarrow$ 2 CuO}
\end{itemize}

\textbf{Aufgabe 4:}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item a) Fehler: O statt O\textsubscript{2} $\rightarrow$ Korrektur: \textcolor{loesung}{4 Al + 3 O\textsubscript{2} $\rightarrow$ 2 Al\textsubscript{2}O\textsubscript{3}}
    \item b) Fehler: Index statt Vorfaktor (MgO\textsubscript{2} $\neq$ MgO) $\rightarrow$ Korrektur: \textcolor{loesung}{2 Mg + O\textsubscript{2} $\rightarrow$ 2 MgO}
\end{itemize}

\textbf{Aufgabe 5:}
\begin{itemize}[leftmargin=1.5em, itemsep=0pt]
    \item Oxidformel (Alg. 1): Na(I), O(II) $\rightarrow$ \textcolor{loesung}{Na\textsubscript{2}O}
    \item Gleichung (Alg. 2): \textcolor{loesung}{4 Na + 1 O\textsubscript{2} $\rightarrow$ 2 Na\textsubscript{2}O}
\end{itemize}
