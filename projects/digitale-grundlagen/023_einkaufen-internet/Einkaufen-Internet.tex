\textbf{3 Merkmale seriöser Shops:}
\begin{enumerate}
\item \textbf{https://} und Schloss-Symbol
\item \textbf{Impressum} mit Adresse \& Telefon
\item \textbf{Bekannte Namen} (Amazon, Otto, etc.)
\end{enumerate}

\textbf{Empfohlene Shops:}
\begin{itemize}
\item \textbf{Amazon} -- Größte Auswahl
\item \textbf{Otto} -- Kauf auf Rechnung
\item \textbf{MediaMarkt/Saturn} -- Elektronik
\item \textbf{dm/Rossmann} -- Drogerie
\item \textbf{Lokale Händler} -- Unterstützt die Region
\end{itemize}

\textbf{5 Warnsignale:}
\begin{enumerate}
\item Preise \textbf{viel zu günstig}
\item \textbf{Nur Vorkasse} als Zahlung
\item \textbf{Kein Impressum}
\item \textbf{Schlechtes Deutsch}
\item \textbf{Nur 5-Sterne-Bewertungen}
\end{enumerate}

\textbf{Beispiel:}\\
{\footnotesize iPhone für 199\euro{} statt 1.199\euro{}? \\
$\rightarrow$ Zu schön, um wahr zu sein = Betrug!}

\textbf{Faustregel:}\\
Ist der Preis zu gut, ist es Betrug!
