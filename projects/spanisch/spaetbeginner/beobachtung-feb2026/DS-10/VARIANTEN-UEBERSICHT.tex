\begin{varbox}[title={\textcolor{varA}{VARIANTE A: Tandem-Interview mit Ankreuz-Vorlage} \hfill \score{9.4} \risikomittel}]{varA}

\en{Variant A: Tandem Interview with Tick-Box Template}

\vspace{0.2cm}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Konzept | \en{Concept}}\\[0.1cm]
Homogene Phase (A↔A, B↔B) → Gemischte Tandems (B interviewt A) → 3 Tandems präsentieren Partner in 3. Person.\\[0.1cm]
\en{Homogeneous phase → Mixed tandems (B interviews A) → 3 tandems present partner in 3rd person.}

\vspace{0.3cm}

\textbf{Verlauf (45 Min) | \en{Procedure}}\\[0.1cm]
\small
\begin{tabular}{|c|l|}
\hline
\rowcolor{varA!10}
\textbf{Min} & \textbf{Phase} \\
\hline
0--3 & Orientierung \\
3--15 & Homogene Interviews \\
15--33 & Gemischte Tandems \\
33--42 & Präsentation (3 Tandems) \\
42--45 & Cierre \\
\hline
\end{tabular}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Checkboxen | \en{Criteria Met}}\\[0.1cm]
\begin{tabular}{ll}
\cmark & Differenzierung (B→A explizit) \\
\cmark & Kommunikation (Interview) \\
\cmark & Scaffolding (Ankreuz-Vorlage) \\
\cmark & 3. Person (Präsentation) \\
\end{tabular}

\vspace{0.3cm}

\textbf{Pro / Kontra}\\[0.1cm]
\begin{tabular}{ll}
\pro & Zeigt Peer-Scaffolding am klarsten \\
\pro & Grammatischer Transfer sichtbar \\
\kontra & Timing bei Präsentation kritisch \\
\kontra & Komplexer Ablauf \\
\end{tabular}

\vspace{0.2cm}

\textbf{Timing-Sicherung | \en{Timing Safety}}\\[0.1cm]
Ankreuz-Vorlage statt freiem Notieren; nur 3 vorher festgelegte Tandems präsentieren.
\end{minipage}

\end{varbox}

\vspace{0.4cm}

\begin{varbox}[title={\textcolor{varB}{VARIANTE B: Speed-Dating} \hfill \score{8.7} \risikoniedrig}]{varB}

\en{Variant B: Speed Dating}

\vspace{0.2cm}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Konzept | \en{Concept}}\\[0.1cm]
6×3 Minuten Kurzinterviews mit festem Partnerwechsel. Kein Notieren während des Gesprächs. Timer steuert alles.\\[0.1cm]
\en{6×3 min short interviews with fixed partner rotation. No note-taking during conversation. Timer controls everything.}

\vspace{0.3cm}

\textbf{Verlauf (45 Min) | \en{Procedure}}\\[0.1cm]
\small
\begin{tabular}{|c|l|}
\hline
\rowcolor{varB!10}
\textbf{Min} & \textbf{Phase} \\
\hline
0--3 & Orientierung \\
3--21 & 6 Runden à 3 Min \\
21--30 & Reflexionsphase \\
30--40 & Plenum: Highlights \\
40--45 & Cierre \\
\hline
\end{tabular}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Checkboxen | \en{Criteria Met}}\\[0.1cm]
\begin{tabular}{ll}
\cmark & Differenzierung (B startet) \\
\cmark & Kommunikation (sehr hoch!) \\
\cmark & Scaffolding (Fragekarten) \\
\xmark & 3. Person (nur implizit) \\
\end{tabular}

\vspace{0.3cm}

\textbf{Pro / Kontra}\\[0.1cm]
\begin{tabular}{ll}
\pro & Timing 100\% sicher \\
\pro & Höchster Sprechanteil \\
\pro & Einfache Struktur \\
\kontra & Kein grammatischer Transfer \\
\kontra & Keine schriftliche Produktion \\
\end{tabular}

\vspace{0.2cm}

\textbf{Timing-Sicherung | \en{Timing Safety}}\\[0.1cm]
Feste 3-Min-Intervalle. Gong/Timer steuert. Nichts kann überlaufen.
\end{minipage}

\end{varbox}

\vspace{0.4cm}

\begin{varbox}[title={\textcolor{varD}{VARIANTE D: Kugellager (Doppelkreis)} \hfill \score{9.3} \risikoniedrig}]{varD}

\en{Variant D: Inside-Outside Circle}

\vspace{0.2cm}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Konzept | \en{Concept}}\\[0.1cm]
Innen- und Außenkreis. B startet innen (interviewt). Nach 3 Runden Rollentausch. Am Ende: 4 SuS präsentieren Partner.\\[0.1cm]
\en{Inner and outer circle. B starts inside (interviewing). Role swap after 3 rounds. Finally: 4 students present partners.}

\vspace{0.3cm}

\textbf{Verlauf (45 Min) | \en{Procedure}}\\[0.1cm]
\small
\begin{tabular}{|c|l|}
\hline
\rowcolor{varD!10}
\textbf{Min} & \textbf{Phase} \\
\hline
0--3 & Aufstellung erklären \\
3--15 & Runde 1--3 (B interviewt) \\
15--17 & Rollentausch \\
17--29 & Runde 4--6 (A interviewt) \\
29--35 & Notizphase \\
35--43 & Präsentation (4 SuS) \\
43--45 & Cierre \\
\hline
\end{tabular}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Checkboxen | \en{Criteria Met}}\\[0.1cm]
\begin{tabular}{ll}
\cmark & Differenzierung (B startet) \\
\cmark & Kommunikation (6 Partner!) \\
\cmark & Scaffolding (B als Modell) \\
\cmark & 3. Person (Präsentation) \\
\end{tabular}

\vspace{0.3cm}

\textbf{Pro / Kontra}\\[0.1cm]
\begin{tabular}{ll}
\pro & Maximaler Sprechanteil \\
\pro & Dynamisch, Bewegung \\
\pro & Alle Boxen getickt \\
\kontra & Braucht Platz im Raum \\
\kontra & Aufstellung dauert \\
\end{tabular}

\vspace{0.2cm}

\textbf{Timing-Sicherung | \en{Timing Safety}}\\[0.1cm]
Feste 4-Min-Intervalle. Rotation auf Signal. Präsentation nur 4 SuS (vorher festgelegt).
\end{minipage}

\end{varbox}

\vspace{0.4cm}

\begin{varbox}[title={\textcolor{varG}{VARIANTE G: Steckbrief-Detektiv} \hfill \score{9.4} \risikoniedrig}]{varG}

\en{Variant G: Profile Detective}

\vspace{0.2cm}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Konzept | \en{Concept}}\\[0.1cm]
Anonyme Steckbriefe schreiben → Mischen → Detektiv-Phase: "Wer ist das?" → Auflösung in 3. Person.\\[0.1cm]
\en{Write anonymous profiles → Shuffle → Detective phase: "Who is this?" → Reveal in 3rd person.}

\vspace{0.3cm}

\textbf{Verlauf (45 Min) | \en{Procedure}}\\[0.1cm]
\small
\begin{tabular}{|c|l|}
\hline
\rowcolor{varG!10}
\textbf{Min} & \textbf{Phase} \\
\hline
0--3 & Erklärung \\
3--11 & Steckbrief schreiben \\
11--13 & Sammeln, mischen, verteilen \\
13--28 & Detektiv-Phase \\
28--40 & Auflösung (3. Person) \\
40--45 & Cierre \\
\hline
\end{tabular}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Checkboxen | \en{Criteria Met}}\\[0.1cm]
\begin{tabular}{ll}
\cmark & Differenzierung (Scaffold für A) \\
\cmark & Kommunikation (echte Fragen!) \\
\cmark & Scaffolding (Steckbrief-Vorlage) \\
\cmark & 3. Person (Auflösung) \\
\end{tabular}

\vspace{0.3cm}

\textbf{Pro / Kontra}\\[0.1cm]
\begin{tabular}{ll}
\pro & Spielcharakter = Motivation \\
\pro & Echte Kommunikation (Ziel!) \\
\pro & Schriftlich + mündlich \\
\kontra & Steckbrief-Phase braucht Zeit \\
\end{tabular}

\vspace{0.2cm}

\textbf{Timing-Sicherung | \en{Timing Safety}}\\[0.1cm]
Steckbrief-Vorlage (nur ausfüllen). Detektiv-Phase flexibel. Auflösung auf 12 Min begrenzt (10 SuS × ~1 Min).
\end{minipage}

\end{varbox}
