\begin{tabular}{@{}p{8.5cm}p{8.5cm}p{8.5cm}@{}}

% --- SER ---
\begin{tcolorbox}[
    colback=ser!10,
    colframe=ser,
    title={\color{white}\Large\bfseries SER},
    fonttitle=\bfseries,
    width=8.3cm,
    height=11cm,
    valign=top
]
\textbf{\large Identität \& Eigenschaften}

\vspace{0.3cm}

\textbf{Verwendung:}
\begin{itemize}
    \item Wer/Was ist jemand?
    \item Dauerhaft, wesentlich
    \item Herkunft, Beruf, Material
\end{itemize}

\vspace{0.3cm}

\textbf{Konjugation:}\\[0.2cm]
\begin{tabular}{ll}
yo & \textbf{soy} \\
tú & \textbf{eres} \\
él/ella & \textbf{es} \\
nosotros & \textbf{somos} \\
vosotros & \textbf{sois} \\
ellos & \textbf{son} \\
\end{tabular}

\vspace{0.3cm}

\textbf{Beispiele:}\\
\textit{María \textbf{es} profesora.}\\
\textit{Yo \textbf{soy} de España.}\\
\textit{La mesa \textbf{es} de madera.}
\end{tcolorbox}

&

% --- ESTAR ---
\begin{tcolorbox}[
    colback=estar!10,
    colframe=estar,
    title={\color{white}\Large\bfseries ESTAR},
    fonttitle=\bfseries,
    width=8.3cm,
    height=11cm,
    valign=top
]
\textbf{\large Zustand \& Ort}

\vspace{0.3cm}

\textbf{Verwendung:}
\begin{itemize}
    \item Wo ist jemand/etwas?
    \item Temporär, veränderlich
    \item Gefühle, Zustände
\end{itemize}

\vspace{0.3cm}

\textbf{Konjugation:}\\[0.2cm]
\begin{tabular}{ll}
yo & \textbf{estoy} \\
tú & \textbf{estás} \\
él/ella & \textbf{está} \\
nosotros & \textbf{estamos} \\
vosotros & \textbf{estáis} \\
ellos & \textbf{están} \\
\end{tabular}

\vspace{0.3cm}

\textbf{Beispiele:}\\
\textit{María \textbf{está} en casa.}\\
\textit{Yo \textbf{estoy} cansado.}\\
\textit{El café \textbf{está} frío.}
\end{tcolorbox}

&

% --- HAY ---
\begin{tcolorbox}[
    colback=hay!10,
    colframe=hay,
    title={\color{white}\Large\bfseries HAY},
    fonttitle=\bfseries,
    width=8.3cm,
    height=11cm,
    valign=top
]
\textbf{\large Existenz \& Vorhandensein}

\vspace{0.3cm}

\textbf{Verwendung:}
\begin{itemize}
    \item Gibt es...? Es gibt...
    \item Unbestimmte Dinge
    \item Immer gleiche Form!
\end{itemize}

\vspace{0.3cm}

\textbf{Form:}\\[0.2cm]
\begin{tabular}{l}
\textbf{hay} (Singular \& Plural)\\[0.3cm]
\textit{= haber (impersonal)}
\end{tabular}

\vspace{0.8cm}

\textbf{Beispiele:}\\
\textit{\textbf{Hay} un parque aquí.}\\
\textit{\textbf{Hay} muchos coches.}\\
\textit{¿\textbf{Hay} una farmacia?}

\vspace{0.3cm}

\footnotesize\textit{Achtung: hay + un/una/muchos}\\
\footnotesize\textit{NICHT: hay el/la}
\end{tcolorbox}

\\
\end{tabular}
