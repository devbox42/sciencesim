    \begin{itemize}[leftmargin=1.5em]
        \item ¿Cómo \textbf{es} tu barrio? \textit{(tranquilo, ruidoso, grande, pequeño...)}
        \item ¿Qué \textbf{hay} en tu barrio? \textit{(tiendas, parques, iglesia, supermercado...)}
        \item ¿Dónde \textbf{está} tu casa? \textit{(cerca de..., al lado de...)}
    \end{itemize}

    \vspace{0.3cm}

    \item {\large\bfseries Partner B:} Stelle 3 Fragen:
    \begin{itemize}[leftmargin=1.5em]
        \item ¿\textbf{Hay}...? \textit{(un supermercado, una farmacia, un parque...)}
        \item ¿Cómo \textbf{es}...? \textit{(la calle, el parque, tu casa...)}
        \item ¿Dónde \textbf{está}...? \textit{(el cine, la parada de autobús...)}
    \end{itemize}

    \vspace{0.3cm}

    \item {\large\bfseries Cambio de roles} \textit{(Rollentausch)}
\end{enumerate}

\begin{center}
{\Large\bfseries\textcolor{gruppeB}{Presentación individual}}\\[0.3cm]
{\large Dauer: 3--4 Minuten (alleine sprechen)}
\end{center}

\vspace{0.5cm}

\hrule
\vspace{0.5cm}

{\Large\bfseries Tu tarea:} Präsentiere deinen Traumberuf. Sprich über:

\vspace{0.5cm}

\begin{enumerate}[label=\textbf{\arabic*.}, leftmargin=2em, itemsep=0.8em]
    \item {\large\bfseries Tu sueño:} Welchen Beruf möchtest du? Warum?
    \begin{itemize}[leftmargin=1.5em]
        \item \textbf{Quiero ser...} \textit{porque...}
        \item \textbf{Me gustaría trabajar como...} \textit{porque...}
    \end{itemize}

    \item {\large\bfseries Las habilidades:} Welche Fähigkeiten brauchst du?
    \begin{itemize}[leftmargin=1.5em]
        \item \textbf{Necesito saber...} / \textbf{Tengo que poder...}
        \item \textbf{Es importante...} \textit{(hablar idiomas, trabajar en equipo...)}
    \end{itemize}

    \item {\large\bfseries El camino:} Wie ist der Weg zu diesem Beruf?
    \begin{itemize}[leftmargin=1.5em]
        \item \textbf{Primero} voy a... / \textbf{Después} tengo que...
        \item \textbf{Finalmente} quiero...
    \end{itemize}
\end{enumerate}

\vspace{0.5cm}
\hrule
\vspace{0.5cm}

{\Large\bfseries Vocabulario útil:}

\vspace{0.3cm}

\begin{tabularx}{\textwidth}{@{}|X|X|@{}}
\hline
\rowcolor{gruppeB!10}
\textbf{Berufe} & \textbf{Verben \& Wendungen} \\
\hline
médico/a, enfermero/a & quiero ser, me gustaría trabajar \\
profesor/a, maestro/a & necesito saber, tengo que poder \\
ingeniero/a, programador/a & voy a estudiar, quiero aprender \\
abogado/a, periodista & es importante, es necesario \\
arquitecto/a, diseñador/a & primero, después, finalmente \\
\hline
\end{tabularx}

\vspace{0.3cm}

\begin{tabularx}{\textwidth}{@{}|X|X|@{}}
\hline
\rowcolor{gruppeB!10}
\textbf{Konjunktionen} & \textbf{Ausbildung} \\
\hline
porque (weil) & la universidad, el instituto \\
para (um zu) & las prácticas, el examen \\
cuando (wenn) & estudiar, trabajar, aprender \\
\hline
\end{tabularx}

\vspace{0.5cm}

\begin{center}
\colorbox{gruppeB!20}{\parbox{0.9\textwidth}{\centering
\textbf{Erlaubt:} Max. 5 Stichpunkte auf einer Karte -- aber \textbf{frei} sprechen!
}}
