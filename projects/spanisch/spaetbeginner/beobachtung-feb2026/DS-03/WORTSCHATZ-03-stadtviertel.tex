\textbf{Mit HAY (es gibt):}
\begin{itemize}
    \item \textit{En mi barrio \textbf{hay} un parque grande.} -- In meinem Viertel gibt es einen großen Park.
    \item \textit{\textbf{Hay} muchas tiendas.} -- Es gibt viele Geschäfte.
    \item \textit{No \textbf{hay} cine.} -- Es gibt kein Kino.
\end{itemize}

\textbf{Mit ESTAR (Lage):}
\begin{itemize}
    \item \textit{El supermercado \textbf{está} cerca de mi casa.} -- Der Supermarkt ist nah bei meinem Haus.
    \item \textit{La farmacia \textbf{está} al lado del banco.} -- Die Apotheke ist neben der Bank.
    \item \textit{¿Dónde \textbf{está} la biblioteca?} -- Wo ist die Bibliothek?
\end{itemize}

\textbf{Mit SER (Eigenschaften):}
\begin{itemize}
    \item \textit{Mi barrio \textbf{es} muy tranquilo.} -- Mein Viertel ist sehr ruhig.
    \item \textit{Las calles \textbf{son} bonitas.} -- Die Straßen sind schön.
\end{itemize}

\end{tcolorbox}
