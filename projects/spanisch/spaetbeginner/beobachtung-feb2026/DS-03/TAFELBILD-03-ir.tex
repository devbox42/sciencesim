\textbf{\large 1. IR + a + Ort}\\[0.2cm]
\textit{Voy \textbf{a} la escuela.} -- Ich gehe zur Schule.\\
\textit{Vamos \textbf{al} parque.} -- Wir gehen zum Park.\\[0.3cm]
\footnotesize\fcolorbox{akzent}{akzent!10}{\textbf{a + el = al} (immer zusammenziehen!)}

\vspace{0.5cm}

\textbf{\large 2. IR + a + Infinitiv (Zukunft)}\\[0.2cm]
\textit{Voy \textbf{a estudiar}.} -- Ich werde lernen.\\
\textit{Vamos \textbf{a comer}.} -- Wir werden essen.\\[0.3cm]
\footnotesize\textit{= nahe Zukunft, Pläne, Absichten}

\vspace{0.5cm}

\textbf{\large 3. IR + en + Transportmittel}\\[0.2cm]
\textit{Voy \textbf{en} autobús.} -- Ich fahre mit dem Bus.\\
\textit{Va \textbf{en} bicicleta.} -- Er/Sie fährt Fahrrad.\\[0.2cm]
\footnotesize\textit{Ausnahme: ir \textbf{a pie} = zu Fuß gehen}

\end{tcolorbox}
