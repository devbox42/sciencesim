% Kartenmaße: 4 Spalten × 3 Zeilen = 12 Karten pro Seite
% A4 = 210mm × 297mm, Margin 10mm → 190mm × 277mm nutzbar
% Karte: 45mm × 88mm (mit 2mm Abstand zwischen Karten)

\newcommand{\platzkarte}[1]{%
\begin{tikzpicture}
    % Rahmen (Schnittlinie)
    \draw[spanisch, line width=1.5pt] (0,0) rectangle (45mm,88mm);

    % Platznummer (groß und fett)
    \node[anchor=north, spanisch, font=\fontsize{28}{32}\selectfont\bfseries]
        at (22.5mm,84mm) {P#1};

    % Namenslinie
    \draw[spanisch, line width=0.8pt] (5mm,68mm) -- (40mm,68mm);
    \node[anchor=north, font=\scriptsize, text=gray] at (22.5mm,67mm) {Name};

    % Punkte-Feld
    \draw[spanisch, line width=0.5pt] (5mm,45mm) rectangle (40mm,58mm);
    \node[anchor=north, font=\small\bfseries, spanisch] at (22.5mm,57mm) {Punkte};
    \node[anchor=center, font=\large] at (22.5mm,50mm) {\rule{15mm}{0.4pt} / \maxpunkte};

    % NP-Feld (Notenpunkte)
    \draw[spanisch, line width=0.5pt] (5mm,28mm) rectangle (40mm,41mm);
    \node[anchor=north, font=\small\bfseries, spanisch] at (22.5mm,40mm) {Note (NP)};
    \node[anchor=center, font=\large] at (22.5mm,33mm) {\rule{15mm}{0.4pt}};

    % QR-Code zum Test (18mm Höhe)
    \node[anchor=south] at (22.5mm,2mm) {\qrcode[height=18mm]{\testurl}};
\end{tikzpicture}%
}
