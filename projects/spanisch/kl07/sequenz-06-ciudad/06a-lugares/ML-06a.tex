\end{aufgabe}

\begin{vokabelkasten}

\noindent

% AUFGABE 4: Satze schreiben (5 Punkte)

\begin{aufgabe}{4}
Schreibe 3 kurze Satze auf Spanisch. Benutze die Orte aus dem Merkkasten. \punktebox{5}
\end{aufgabe}

\begin{hinweis}
Beispiel: \es{En mi ciudad hay un parque.} (In meiner Stadt gibt es einen Park.)
\end{hinweis}

\vspace{0.5em}

\textbf{Beispiellosungen:}

\noindent
\renewcommand{\arraystretch}{2.0}
\begin{tabular}{p{\textwidth}}

b) \loesung{La biblioteca está cerca del parque.} \\
c) \loesung{El hospital es grande.} \\
\end{tabular}

\vspace{0.5em}
\textit{\textcolor{grau}{Bewertung: Ort korrekt (1P), Artikel korrekt (0,5P), Satzstruktur korrekt (0,5P) = max. 2P pro Satz}}
