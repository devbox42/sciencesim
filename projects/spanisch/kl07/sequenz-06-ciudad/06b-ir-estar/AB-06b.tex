\end{merkkasten}

\hline
2. Pl. & vosotros & \luecke{2cm} \\
\hline
3. Pl. & ellos/ellas & \luecke{2cm} \\
\hline
\end{tabular}
\end{center}

\vspace{0.5em}
\textbf{Beispiel:} El cine \es{está} en la plaza. = Das Kino ist auf dem Platz.
\end{merkkasten}
