\begin{merkkasten}[Merkkasten: Das Verb \textit{tener} (haben)]

\begin{center}
\begin{tabular}{|l|l|l|}
\hline
\textbf{Person} & \textbf{Pronomen} & \textbf{tener} \\
\hline
1. Sg. & yo (ich) & \loesung{tengo} \\
\hline
2. Sg. & tú (du) & \loesung{tienes} \\
\hline
3. Sg. & él/ella (er/sie) & \loesung{tiene} \\
\hline
1. Pl. & nosotros/-as (wir) & \loesung{tenemos} \\
\hline
\end{tabular}
\end{center}

\vspace{0.5em}

\textbf{Alter angeben:} \es{¿Cuántos años tienes?} = Wie alt bist du? (wörtlich: Wie viele Jahre hast du?)

\vspace{0.3em}

$\rightarrow$ \es{Tengo ... años.} = Ich bin ... Jahre alt. (wörtlich: Ich habe ... Jahre.)

\end{merkkasten}

\vspace{10pt}

% AUFGABE 1: tener einsetzen (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{1}
Setze die richtige Form von \textit{tener} ein. \punktebox{4}
\end{aufgabe}

\begin{enumerate}[label=\alph*)]
    \item Yo \loesung{tengo} 12 años.
    \item Mi hermana \loesung{tiene} 15 años.
    \item ¿Cuántos años \loesung{tienes} tú?
    \item Nosotros \loesung{tenemos} un perro.
\end{enumerate}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 3: Alter angeben (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{3}
Beantworte die Fragen auf Spanisch. \punktebox{6}
\end{aufgabe}

\begin{hinweis}
\textbf{Struktur:} Tengo/Tiene + [Zahl] + años.
\end{hinweis}

\vspace{0.5em}

\noindent
a) \es{¿Cuántos años tienes?}

\loesung{Tengo [12/13/14...] años.}

\vspace{0.5em}

\noindent
b) \es{¿Cuántos años tiene tu madre?}

\loesung{Mi madre tiene [z.B. 42] años.}

\vspace{0.5em}

\noindent
c) \es{¿Cuántos años tiene tu hermano/hermana?}

\loesung{Mi hermano/hermana tiene [z.B. 10] años.}

\vspace{0.5em}

\textit{\textcolor{grau}{Bewertung: Je 2P (1P für korrekte tener-Form, 1P für Zahl + años)}}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 4: Familie beschreiben (mit Beispiellösung)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{4}
Beschreibe deine Familie. Schreibe mindestens 3 Sätze mit Namen und Alter. \punktebox{6}
\end{aufgabe}

\textbf{Beispiellösung:}

\loesung{Mi padre se llama Carlos. Tiene 45 años.}

\loesung{Mi madre se llama María. Tiene 42 años.}

\loesung{Mi hermana se llama Ana. Tiene 10 años.}

\loesung{Yo tengo 13 años.}

\vspace{0.5em}

\textit{\textcolor{grau}{Bewertungskriterien:}}
\begin{itemize}
    \item Mindestens 3 Familienmitglieder genannt: 2P
    \item Korrekte tener-Formen: 2P
    \item Korrekte Zahlen + años: 2P
\end{itemize}
