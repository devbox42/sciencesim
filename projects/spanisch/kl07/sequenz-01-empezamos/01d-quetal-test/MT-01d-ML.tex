\noindent\textbf{Zusammenfassung Sequenz 1: Empezamos}

\begin{itemize}
    \item \textbf{Begrüßung:} Hola, Buenos días, Buenas tardes/noches, Adiós, Hasta luego/mañana/pronto
    \item \textbf{Vorstellen:} Me llamo..., Soy de..., ¿Cómo te llamas?, ¿De dónde eres?
    \item \textbf{ser:} soy, eres, es, somos
    \item \textbf{Artikel:} el/la (Singular), los/las (Plural)
    \item \textbf{Befinden:} ¿Qué tal?, ¿Cómo estás?, Bien/Muy bien/Regular/Mal, ¿Y tú?
\end{itemize}
