\vspace{3pt}

\begin{enumerate}[label=\alph*), nosep, itemsep=4pt]

    \item los zapatos negr\lsg{os}
    \item el jersey azul \lsg{(keine Endung -- 2-Endungen-Farbe)}
    \item las botas marron\lsg{es}
    \item la falda ros\lsg{a (unveränderlich!)}
\end{enumerate}

\vspace{8pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 3: Sätze mit llevar
% ═══════════════════════════════════════════════════════════════

\textbf{\textcolor{\fachfarbe}{Aufgabe 3}} -- Was trägt María? Schreibe Sätze mit \es{llevar}. \punktebox{6}

\vspace{3pt}

\begin{hinweis}
\small \textbf{Beispiel:} (T-Shirt / rot) $\rightarrow$ \es{María lleva una camiseta roja.}
\end{hinweis}

\vspace{3pt}

\begin{enumerate}[label=\alph*), nosep, itemsep=4pt]
    \item (Hose / blau) $\rightarrow$ \lsg{María lleva unos pantalones azules.}
    \item (Jacke / schwarz) $\rightarrow$ \lsg{María lleva una chaqueta negra.}
    \item (Schuhe / weiß) $\rightarrow$ \lsg{María lleva unos zapatos blancos.}
    \item (Rock / grün) $\rightarrow$ \lsg{María lleva una falda verde.}
    \item (Pullover / grau) $\rightarrow$ \lsg{María lleva un jersey gris.}
    \item (Stiefel / braun) $\rightarrow$ \lsg{María lleva unas botas marrones.}
\end{enumerate}

\vspace{8pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 4: Preise
% ═══════════════════════════════════════════════════════════════

\textbf{\textcolor{\fachfarbe}{Aufgabe 4}} -- Ergänze \es{cuesta} oder \es{cuestan}. \punktebox{4}

\vspace{3pt}

\begin{enumerate}[label=\alph*), nosep, itemsep=4pt]
    \item ¿Cuánto \lsg{cuesta} la camiseta? -- \lsg{Cuesta} 15 euros.
    \item ¿Cuánto \lsg{cuestan} los zapatos? -- \lsg{Cuestan} 50 euros.
    \item ¿Cuánto \lsg{cuesta} el vestido? -- \lsg{Cuesta} 35 euros.
    \item ¿Cuánto \lsg{cuestan} las botas? -- \lsg{Cuestan} 60 euros.
\end{enumerate}

\vspace{8pt}
