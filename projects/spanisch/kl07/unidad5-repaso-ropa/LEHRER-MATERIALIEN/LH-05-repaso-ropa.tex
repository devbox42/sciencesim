\section*{5. Tipps für die Durchführung}

\begin{tippbox}[Praxistipps]
\begin{itemize}[nosep, leftmargin=*]
    \item \textbf{Bingo:} Preise für Gewinner (Hausaufgaben-Gutschein, kleine Süßigkeit)
    \item \textbf{Kettenübung:} Langsam starten, Tempo steigern. Ball oder Stofftier weitergeben.
    \item \textbf{Modenschau:} Freiwillige zuerst, dann Gruppen. Musik im Hintergrund motiviert!
    \item \textbf{Zeitpuffer:} Modenschau kann gekürzt werden, wenn Zeit knapp
    \item \textbf{Echte Kleidung:} Wenn möglich, Requisiten mitbringen (Hüte, Schals)
\end{itemize}
\end{tippbox}
