\abschnitt{3}{Llevar} \de{-- Tragen}

\vspace{4pt}

\begin{grammatikbox}[\gram{Konjugation}: llevar]
\footnotesize
\begin{tabular}{@{}ll@{}}
yo \es{llevo} & nosotros \es{llevamos} \\
tú \es{llevas} & vosotros \es{lleváis} \\
él/ella \es{lleva} & ellos/ellas \es{llevan} \\
\end{tabular}

\vspace{4pt}
\textbf{Beispiel:}\\
\es{Hoy llevo una camiseta roja.}\\
\de{Heute trage ich ein rotes T-Shirt.}
\end{grammatikbox}

\vspace{6pt}

\begin{grammatikbox}[\gram{Farbadjektive} angleichen]
\footnotesize
\textbf{4-Endungen:} rojo $\rightarrow$ roj\textbf{o/a/os/as}\\
\es{el jersey rojo} / \es{la falda roja}\\
\es{los zapatos rojos} / \es{las botas rojas}

\vspace{4pt}
\textbf{2-Endungen:} azul, verde, gris, marrón\\
\es{el vestido azul} / \es{los vestidos azules}

\vspace{4pt}
\textbf{Unveränderlich:} rosa, naranja, violeta
\end{grammatikbox}

\abschnitt{5}{De compras} \de{-- Beim Einkaufen}

\vspace{4pt}

\begin{multicols}{2}

\begin{grammatikbox}[Redemittel: Einkaufen]
\footnotesize
\textbf{Verkäufer/in:}
\begin{itemize}[leftmargin=*, nosep, itemsep=0pt]
    \item \es{¿Qué desea?} -- \de{Was wünschen Sie?}
    \item \es{¿Qué talla necesita?} -- \de{Welche Größe?}
    \item \es{Cuesta... euros.} -- \de{Es kostet...}
\end{itemize}

\vspace{3pt}
\textbf{Kunde/in:}
\begin{itemize}[leftmargin=*, nosep, itemsep=0pt]
    \item \es{Quiero comprar...} -- \de{Ich möchte kaufen...}
    \item \es{¿Cuánto cuesta?} -- \de{Wie viel kostet es?}
    \item \es{Me lo/la llevo.} -- \de{Ich nehme es.}
\end{itemize}
\end{grammatikbox}

\columnbreak

{\footnotesize
\textbf{Beispiel-Dialog:}\\[3pt]
\textbf{V:} \es{¡Buenos días! ¿Qué desea?}\\
\textbf{K:} \es{Quiero comprar una camiseta.}\\
\textbf{V:} \es{¿De qué color?}\\
\textbf{K:} \es{Azul, por favor.}\\
\textbf{V:} \es{Esta camiseta cuesta 12 euros.}\\
\textbf{K:} \es{Vale, me la llevo.}}

\end{multicols}
