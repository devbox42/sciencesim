        \item \es{\_\_\_\_\_ muchos libros aquí.}

\begin{frame}{Repaso: Lösungen}
    \vspace{0.5cm}

    \begin{enumerate}
        \setlength{\itemsep}{0.5cm}
        \item \textcolor{richtig}{\textbf{Hay}} veinticinco alumnos. \de{es gibt}
        \item El libro \textcolor{richtig}{\textbf{está}} en la mesa. \de{ist/befindet sich}
        \item \textcolor{richtig}{\textbf{Hay}} muchos libros aquí. \de{es gibt}
        \item ¿Dónde \textcolor{richtig}{\textbf{está}} la profesora? \de{ist/befindet sich}
    \end{enumerate}

    \vspace{0.8cm}

    \begin{block}{Merke}
        \textbf{hay} = es gibt (Existenz) · \textbf{estar} = sein/sich befinden (Ort)
    \end{block}
\end{frame}

\begin{frame}{de + artículo \de{von + Artikel}}
    \vspace{0.3cm}

    {\large\es{¿De quién es?} \de{Wem gehört das?}}

    \vspace{0.6cm}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Kontraktion (nur 1 Fall!)}

            \vspace{0.3cm}

            de + el = \textcolor{akzent}{\textbf{del}}

            \vspace{0.3cm}

            {\small
            \es{el libro \textcolor{akzent}{del} alumno}\\
            \de{das Buch des Schülers}
            }
        \end{column}

        \begin{column}{0.48\textwidth}
            \textbf{Keine Kontraktion}

            \vspace{0.3cm}

            de + la = de la\\
            de + los = de los\\
            de + las = de las

            \vspace{0.3cm}

            {\small
            \es{la silla \textcolor{akzent}{de la} profesora}\\
            \de{der Stuhl der Lehrerin}
            }
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Beispiele \de{Ejemplos}}
    \vspace{0.5cm}

    \renewcommand{\arraystretch}{1.6}
    \begin{tabular}{lll}
        \toprule
        \textbf{Spanisch} & \textbf{Artikel} & \textbf{Deutsch} \\
        \midrule
        el libro \textcolor{akzent}{\textbf{del}} alumno & de + el & das Buch des Schülers \\
        la silla \textcolor{akzent}{\textbf{de la}} profesora & de + la & der Stuhl der Lehrerin \\
        los cuadernos \textcolor{akzent}{\textbf{de los}} alumnos & de + los & die Hefte der Schüler \\
        la mochila \textcolor{akzent}{\textbf{de las}} alumnas & de + las & der Rucksack der Schülerinnen \\
        \bottomrule
    \end{tabular}

    \vspace{0.8cm}

    \begin{block}{¡Recuerda! \de{Merke!}}
        Nur \textbf{de + el} wird zu \textbf{del}. Alle anderen bleiben getrennt!
    \end{block}
\end{frame}

\begin{frame}{Juego: ¿De quién es? \de{Spiel: Wem gehört das?}}
    \vspace{0.3cm}

    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Regeln:}
            \begin{enumerate}
                \item Ich zeige einen Gegenstand
                \item Ihr ratet: \es{Es el/la ... de [Name].}
                \item Der Besitzer sagt: \es{¡Sí!} oder \es{No.}
                \item Richtig geraten? $\rightarrow$ Du bist dran!
            \end{enumerate}
        \end{column}

        \begin{column}{0.4\textwidth}
            \textbf{Sprachgerüst:}

            \vspace{0.3cm}

            {\small
            \es{Es el bolígrafo de María.}\\
            \de{Das ist der Kuli von María.}

            \vspace{0.3cm}

            \es{Es la goma del profesor.}\\
            \de{Das ist der Radiergummi des Lehrers.}

            \vspace{0.3cm}

            \es{¡Sí, es mi bolígrafo!}\\
            \de{Ja, das ist mein Kuli!}
            }
        \end{column}
    \end{columns}

    \vspace{0.5cm}
    \zeit{15 Minuten}
