\begin{frame}{Verbos en -ar: Konjugation}
    \vspace{0.3cm}

    \begin{columns}[T]
        \begin{column}{0.45\textwidth}
            \textbf{Endungen}

            \vspace{0.3cm}

            \renewcommand{\arraystretch}{1.4}
            \begin{tabular}{ll}
                yo & -\endung{o} \\
                tú & -\endung{as} \\
                él/ella & -\endung{a} \\
                nosotros & -\endung{amos} \\
                vosotros & -\endung{áis} \\
                ellos/ellas & -\endung{an} \\
            \end{tabular}
        \end{column}

        \begin{column}{0.5\textwidth}
            \textbf{Beispiel: hablar} \de{sprechen}

            \vspace{0.3cm}

            \renewcommand{\arraystretch}{1.4}
            \begin{tabular}{ll}
                yo & habl\endung{o} \\
                tú & habl\endung{as} \\
                él/ella & habl\endung{a} \\
                nosotros & habl\endung{amos} \\
                vosotros & habl\endung{áis} \\
                ellos/ellas & habl\endung{an} \\
            \end{tabular}
        \end{column}
    \end{columns}

    \vspace{0.5cm}

    \begin{block}{Tipp}
        Stamm = Infinitiv ohne -ar (habl-), dann Endung anhängen!
    \end{block}
\end{frame}
