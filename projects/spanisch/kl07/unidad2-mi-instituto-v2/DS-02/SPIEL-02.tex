\begin{regelbox}
\textbf{\large Spielregeln} \de{-- Reglas del juego}

\vspace{6pt}

\begin{enumerate}[leftmargin=2em, itemsep=4pt]
    \item \textbf{Ein Schüler schließt die Augen.}\\
    \de{Un alumno cierra los ojos.}

    \item \textbf{Die Klasse versteckt einen Gegenstand} (sichtbar, aber ungewöhnlicher Ort).\\
    \de{La clase esconde un objeto (visible, pero en un lugar inusual).}

    \item \textbf{Der Schüler fragt:} \estar{¿Dónde está el/la...?}\\
    \de{El alumno pregunta: ¿Dónde está...?}

    \item \textbf{Die Klasse gibt Hinweise mit Präpositionen.}\\
    \de{La clase da pistas con preposiciones.}

    \item \textbf{Nach dem Finden: vollständigen Satz sagen!}\\
    \de{Después de encontrarlo: ¡decir la frase completa!}
\end{enumerate}
\end{regelbox}

\vspace{8pt}
