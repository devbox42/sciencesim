\begin{merkkasten}[Merkkasten: hay (es gibt)]
\textbf{Die Struktur \es{hay}:}

\vspace{0.5em}

\es{Hay} bedeutet ``\loesung{es gibt}'' und wird für Singular \textbf{und} Plural verwendet.

\vspace{0.8em}

\textbf{Struktur:}

\begin{center}
\fbox{\es{hay} + \es{un/una} + Substantiv}
\end{center}

\vspace{0.5em}

\textbf{Beispiele:}

\begin{tabular}{ll}
$\bullet$ En mi dormitorio \es{hay una} cama. & = \loesung{In meinem Schlafzimmer gibt es ein Bett.} \\[0.6em]
$\bullet$ En el salón \es{hay un} sofá. & = \loesung{Im Wohnzimmer gibt es ein Sofa.} \\
\end{tabular}

\vspace{0.8em}

\textbf{ACHTUNG:} Nach \es{hay} steht der \textbf{un}bestimmte Artikel (un/una), \textbf{nicht} el/la!

\end{merkkasten}

\vspace{10pt}

% ═══════════════════════════════════════════════════════════════
% MERKKASTEN 2: Möbelvokabeln (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{merkkasten}[Merkkasten: Los muebles (Die Möbel)]

\vspace{0.5em}

\begin{tabular}{ll|ll}
\es{la mesa} & = \loesung{der Tisch} & \es{el armario} & = \loesung{der Schrank} \\[0.8em]
\es{la silla} & = \loesung{der Stuhl} & \es{la lámpara} & = \loesung{die Lampe} \\[0.8em]
\es{el sofá} & = \loesung{das Sofa} & \es{la estantería} & = \loesung{das Regal} \\[0.8em]
\es{la cama} & = \loesung{das Bett} & \es{el escritorio} & = \loesung{der Schreibtisch} \\
\end{tabular}

\vspace{0.5em}

\textit{Tipp: Lerne die Wörter immer mit Artikel! (el = maskulin, la = feminin)}

\end{merkkasten}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 1: Möbel zuordnen (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{1: Möbel zuordnen}
Verbinde das spanische Wort mit der deutschen Bedeutung. \punktebox{4}
\end{aufgabe}

\begin{center}
\begin{tabular}{rcl}
\es{la cama} & \loesung{---} & der Schrank \\[0.4em]
& & \loesung{$\rightarrow$ la cama = das Bett} \\[0.8em]
\es{el escritorio} & \loesung{---} & das Bett \\[0.4em]
& & \loesung{$\rightarrow$ el escritorio = der Schreibtisch} \\[0.8em]
\es{la estantería} & \loesung{---} & der Schreibtisch \\[0.4em]
& & \loesung{$\rightarrow$ la estantería = das Regal} \\[0.8em]
\es{el armario} & \loesung{---} & das Regal \\[0.4em]
& & \loesung{$\rightarrow$ el armario = der Schrank} \\
\end{tabular}
\end{center}

\textit{\textcolor{grau}{Bewertung: 1 Punkt pro richtige Zuordnung}}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 2: hay + un/una einsetzen (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{2: hay + un/una einsetzen}
Ergänze die Sätze mit \es{hay un} oder \es{hay una}. \punktebox{4}
\end{aufgabe}

\begin{vokabelkasten}
\textit{Zur Erinnerung:} un = maskulin (el) | una = feminin (la)
\end{vokabelkasten}

\vspace{0.5em}

\noindent
a) En mi dormitorio \loesung{hay una} cama grande.

\vspace{0.6em}
\noindent
b) En el salón \loesung{hay un} sofá nuevo.

\vspace{0.6em}
\noindent
c) En la cocina \loesung{hay una} mesa.

\vspace{0.6em}
\noindent
d) En mi habitación \loesung{hay un} escritorio.

\vspace{0.5em}
\textit{\textcolor{grau}{Bewertung: 1 Punkt pro richtigen Artikel (una bei la-Wörtern, un bei el-Wörtern)}}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 3: Zimmer beschreiben (mit Lösungen)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{3: Zimmer beschreiben}
Was gibt es in diesen Zimmern? Schreibe Sätze mit ``En... hay...''. \punktebox{6}
\end{aufgabe}

\begin{hinweis}
Beginne jeden Satz mit dem Zimmer: \es{En el dormitorio}, \es{En el salón}, \es{En la cocina}...
\end{hinweis}

\vspace{0.8em}

\noindent
\textbf{Beispiel:} \textit{Schlafzimmer: Bett, Schrank}

$\rightarrow$ \es{En el dormitorio hay una cama y un armario.}

\vspace{1em}

\noindent
a) \textbf{Wohnzimmer:} Sofa, Lampe

\vspace{0.3em}
\noindent
\loesung{En el salón hay un sofá y una lámpara.}

\vspace{1em}

\noindent
b) \textbf{Küche:} Tisch, 4 Stühle

\vspace{0.3em}
\noindent
\loesung{En la cocina hay una mesa y cuatro sillas.}

\vspace{1em}

\noindent
c) \textbf{Arbeitszimmer:} Schreibtisch, Regal, Lampe

\vspace{0.3em}
\noindent
\loesung{En el despacho hay un escritorio, una estantería y una lámpara.}

\vspace{0.5em}
\textit{\textcolor{grau}{Bewertung: 2 Punkte pro Satz (1P Struktur, 1P Vokabeln). Auch ``En mi habitación'' oder ``En la oficina'' akzeptabel.}}

\vspace{15pt}

% ═══════════════════════════════════════════════════════════════
% AUFGABE 4: Eigenes Zimmer beschreiben (Beispiellösung)
% ═══════════════════════════════════════════════════════════════

\begin{aufgabe}{4: Dein Zimmer beschreiben}
Beschreibe dein eigenes Zimmer auf Spanisch. Schreibe mindestens 4 Sätze mit \es{hay}. \punktebox{6}
\end{aufgabe}

\textbf{Beispiellösung:}

\vspace{0.5em}

\loesung{Mi habitación es pequeña pero bonita.}

\loesung{Hay una cama grande.}

\loesung{Hay un armario para mi ropa.}

\loesung{Hay un escritorio con una lámpara.}

\loesung{Hay una estantería con muchos libros.}

\loesung{No hay sofá en mi habitación.}

\vspace{1em}

\textit{\textcolor{grau}{Bewertungskriterien:}}
\begin{itemize}
    \item \textcolor{grau}{Mindestens 4 Sätze mit hay: 2P}
    \item \textcolor{grau}{Korrekter Artikel (un/una): 2P}
    \item \textcolor{grau}{Korrekte Möbelvokabeln: 1P}
    \item \textcolor{grau}{Sprachliche Korrektheit (Rechtschreibung, Akzente): 1P}
\end{itemize}
