\begin{tabularx}{\textwidth}{|l|X|X|}

\textbf{Fehler} & \textbf{Beispiel} & \textbf{Reaktion} \\

\hline
Verwechslung manzana/naranja & Ähnliche Endung & Farbcode: manzana = grün/rot, naranja = orange \\
\hline
\end{tabularx}
\end{fehlerbox}
