\begin{aufgabe}{3 -- Setze die richtige Form ein}
Ergänze die richtige Form von \es{querer} oder \es{poder}. \punktebox{8}
\end{aufgabe}

\noindent
a) Yo \lsg{quiero} comprar manzanas. \hfill (querer -- ich möchte)

\vspace{0.8em}

\noindent
b) ¿Tú \lsg{puedes} ayudarme? \hfill (poder -- kannst du)

\vspace{0.8em}

\noindent
c) María \lsg{quiere} un kilo de tomates. \hfill (querer -- sie möchte)

\vspace{0.8em}

\noindent
d) Nosotros no \lsg{podemos} pagar con tarjeta. \hfill (poder -- wir können nicht)

\vspace{0.8em}

\noindent
e) Ellos \lsg{quieren} ir al mercado. \hfill (querer -- sie wollen)

\vspace{0.8em}

\noindent
f) ¿Vosotros \lsg{podéis} hablar español? \hfill (poder -- könnt ihr)

\vspace{0.8em}

\noindent
g) Usted \lsg{quiere} un café, ¿verdad? \hfill (querer -- Sie möchten)

\vspace{0.8em}

\noindent
h) Yo no \lsg{puedo} entender. \hfill (poder -- ich kann nicht)

\vspace{15pt}

\begin{aufgabe}{4 -- Übersetze ins Spanische}

\noindent
a) Ich möchte Orangen kaufen.

\lsg{Quiero comprar naranjas.}

\vspace{1em}

\noindent

\lsg{¿Puedes ayudarme?} \textit{(oder: ¿Me puedes ayudar?)}

\vspace{1em}

\noindent
c) Wir können nicht bezahlen.

\lsg{No podemos pagar.}
