% --- VOR DER PRÜFUNG ---
\textbf{\large Vor der Prüfung}

\begin{tippbox}
\tipp{1. Laut üben!}\\[2pt]
Sprich deine Texte \textbf{laut} -- nicht nur im Kopf. Nur so merkst du, wo du stockst.
\end{tippbox}

\begin{tippbox}
\tipp{2. Aufnehmen \& anhören}\\[2pt]
Nimm dich mit dem Handy auf. Höre dir die Aufnahme an: Wo sind Pausen? Wo klingt es unsicher?
\end{tippbox}

\begin{tippbox}
\tipp{3. Vor dem Spiegel}\\[2pt]
Übe vor dem Spiegel. Achte auf deine Körpersprache und Mimik.
\end{tippbox}

\begin{tippbox}
\tipp{4. Mit Partner üben}\\[2pt]
Übe mit einem Mitschüler oder Familienmitglied. Lass dir Fragen stellen!
\end{tippbox}

\begin{tippbox}
\tipp{5. Struktur merken}\\[2pt]
Lerne keine Texte auswendig, sondern eine \textbf{Struktur}:\\
\es{Presentación → Familia → Barrio}
\end{tippbox}

\columnbreak

% --- WÄHREND DER PRÜFUNG ---
\textbf{\large Während der Prüfung}

\begin{tippbox}
\tipp{6. Ruhig bleiben}\\[2pt]
Atme tief durch. Nervosität ist normal! Eine kurze Pause ist besser als Stottern.
\end{tippbox}

\begin{tippbox}
\tipp{7. Langsam sprechen}\\[2pt]
Sprich \textbf{langsam und deutlich}. Das gibt dir Zeit zum Denken und klingt sicherer.
\end{tippbox}

\begin{tippbox}
\tipp{8. Nachfragen erlaubt!}\\[2pt]
Wenn du eine Frage nicht verstehst:\\[2pt]
\es{¿Puede repetir, por favor?}\\
\es{¿Cómo? / ¿Perdón?}
\end{tippbox}

\begin{tippbox}
\tipp{9. Einfach anfangen}\\[2pt]
Beginne mit etwas Einfachem:\\[2pt]
\es{¡Hola! Me llamo... y tengo... años.}\\[2pt]
Der Anfang gibt dir Sicherheit!
\end{tippbox}

\begin{tippbox}
\tipp{10. Fehler ignorieren}\\[2pt]
Hast du einen Fehler gemacht? \textbf{Weitersprechen!} Nicht korrigieren, nicht stocken.
\end{tippbox}

\end{multicols}

\vspace{8pt}

% --- WICHTIGSTER TIPP ---
\begin{successbox}
\begin{center}
\textbf{\large Der wichtigste Tipp:}\\[6pt]
{\Large \textbf{Sprich!}}\\[4pt]
Ein Satz mit kleinem Fehler ist besser als Schweigen.\\[2pt]
Zeige, was du kannst -- nicht, was du nicht kannst!
\end{center}
\end{successbox}

\vspace{8pt}

% --- NÜTZLICHE PHRASEN ---
\begin{center}
\textbf{Nützliche Phrasen für die Prüfung}
\end{center}

\vspace{4pt}

\begin{multicols}{2}
\footnotesize

\textbf{Zeit gewinnen:}\\
\es{Bueno...} \textit{-- Also...}\\
\es{A ver...} \textit{-- Mal sehen...}\\
\es{Pues...} \textit{-- Also/Nun...}\\
\es{Es que...} \textit{-- Es ist so, dass...}

\vspace{6pt}

\textbf{Nachfragen:}\\
\es{¿Puede repetir, por favor?}\\
\es{¿Cómo?} / \es{¿Perdón?}\\
\es{No entiendo la pregunta.}

\columnbreak

\textbf{Unsicherheit ausdrücken:}\\
\es{Creo que...} \textit{-- Ich glaube, dass...}\\
\es{No estoy seguro/a, pero...}\\
\es{Quizás...} \textit{-- Vielleicht...}

\vspace{6pt}

\textbf{Abschluss:}\\
\es{Eso es todo.} \textit{-- Das ist alles.}\\
\es{No sé más.} \textit{-- Mehr weiß ich nicht.}

\end{multicols}
