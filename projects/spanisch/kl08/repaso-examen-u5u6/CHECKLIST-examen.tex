\textbf{\textcolor{spanisch}{2. Ropa + Adjetivos (Adjektivanpassung)}} \hfill \checkbox Kann ich!

\small
Adjektive stehen \textbf{nach} dem Substantiv und passen sich an:
\begin{itemize}[leftmargin=15pt, itemsep=1pt]
    \item \es{el jersey rojo} (mask. Sg.) / \es{la falda roja} (fem. Sg.)
    \item \es{los pantalones negros} (mask. Pl.) / \es{las camisetas azules} (fem. Pl.)
    \item \textbf{Achtung:} azul, verde, gris, marrón, rosa, naranja -- unveränderlich!
\end{itemize}

\small
\begin{tabular}{l|ccc}

\hline

\textbf{\textcolor{spanisch}{5. Números 1-1000}} \hfill \checkbox Kann ich!

\small
\begin{tabular}{llllll}
100 & \es{cien} & 400 & \es{cuatrocientos/as} & 700 & \es{setecientos/as} \\
200 & \es{doscientos/as} & 500 & \es{quinientos/as} & 800 & \es{ochocientos/as} \\
300 & \es{trescientos/as} & 600 & \es{seiscientos/as} & 900 & \es{novecientos/as} \\
& & & & 1000 & \es{mil} \\
\end{tabular}

\vspace{3pt}
\textbf{Regeln:}
\begin{itemize}[leftmargin=15pt, itemsep=0pt]
    \item 100 allein = \es{cien} | 100 + Zahl = \es{ciento} (ciento uno = 101)
    \item \textbf{Unregelmäßig:} \es{quinientos} (500), \es{setecientos} (700), \es{novecientos} (900)
    \item \textbf{Genus:} doscientos euros / doscient\textbf{as} personas
\end{itemize}

\end{themabox}

\begin{themabox}[UNIDAD 5: Verbo ir]

\textbf{\textcolor{spanisch}{6. Verbo ir + a + el/la}} \hfill \checkbox Kann ich!

\small
\begin{tabular}{ll|ll}
yo & \es{voy} & nosotros & \es{vamos} \\
tú & \es{vas} & vosotros & \es{vais} \\
él/ella & \es{va} & ellos/ellas & \es{van} \\
\end{tabular}

\vspace{3pt}
\textbf{Kontraktion:} \es{a + el = al} (immer!) | a + la = a la (keine Veränderung)
\begin{itemize}[leftmargin=15pt, itemsep=0pt]
    \item Voy \es{al} cine. (a + el) | Voy \es{a la} tienda. (a + la)
\end{itemize}
