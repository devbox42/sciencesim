Auf Spanisch kannst du \textbf{nicht} sagen: „Ich mag dich." Stattdessen sagst du: \es{Me gustas.} -- wörtlich: „Du gefällst mir." Das Subjekt ist immer das, \textbf{was gefällt} -- nie die Person, die mag!

% ---------------------------------------------------------------
% ABSCHNITT 1: STRUKTUR
% ---------------------------------------------------------------
\abschnitt{1}{Gustar = Gefallen}

\vspace{4pt}

\begin{grammatikbox}
\textbf{Der entscheidende Unterschied}

\vspace{4pt}

\begin{tabular}{@{}l@{ }l@{}}
\textbf{Falsch:} & \es{*Yo gusto música.} \\
& \de{(wie „mögen")} \\[4pt]
\textbf{Richtig:} & \es{Me gusta la música.} \\
& \de{Mir gefällt die Musik.} \\
\end{tabular}

\vspace{6pt}

\textbf{Struktur:}\\
\es{Me} = Wem? (Dativ)\\
\es{la música} = Was gefällt? (\textbf{Subjekt!})
\end{grammatikbox}

\vspace{4pt}

\begin{fehlerbox}
{\small\textbf{Typische Fehler:}}\\[2pt]
{\footnotesize
\textcolor{fehler}{*Yo gusto...} $\rightarrow$ Falsche Struktur!\\
\textcolor{fehler}{*Me gusta los libros.} $\rightarrow$ Plural: \es{gustan}!\\
\textcolor{fehler}{*Me gustan bailar.} $\rightarrow$ Infinitiv: \es{gusta}!}
\end{fehlerbox}

\vspace{4pt}

\abmark{Kasten 1}

\vspace{8pt}

% ---------------------------------------------------------------
% ABSCHNITT 2: PRONOMEN
% ---------------------------------------------------------------
\abschnitt{2}{Los pronombres} \de{-- Die Pronomen}

\vspace{4pt}

{\small
\begin{tabular}{@{}lll@{}}
\textbf{Person} & \textbf{Pronomen} & \textbf{Deutsch} \\
\midrule
(a mí) & \es{me} & mir \\
(a ti) & \es{te} & dir \\
(a él/ella) & \es{le} & ihm/ihr \\
(a nosotros) & \es{nos} & uns \\
(a vosotros) & \es{os} & euch \\
(a ellos) & \es{les} & ihnen \\
\end{tabular}}

\vspace{6pt}

\begin{merkbox}
\textbf{Kernregel:}\\
\es{gusta} + Singular / Infinitiv\\
\es{gustan} + Plural
\end{merkbox}

\vspace{4pt}

\begin{tippbox}
{\footnotesize\textit{Wozu „a mí", „a ti"...?}\\
Optional -- aber nützlich für \textbf{Betonung} (\es{A mí} me gusta, \es{a ti} no.) und \textbf{Klarstellung} bei le/les.}
\end{tippbox}

\vspace{4pt}

\abmark{Kasten 2 + Übung 1}

\abschnitt{3}{Ejemplos} \de{-- Beispiele}

{\small
\begin{tabular}{@{}l@{ }l@{}}
\es{Me gusta bailar.} & \de{Mir gefällt tanzen.} \\[3pt]
\es{Te gustan los libros.} & \de{Dir gefallen die Bücher.} \\[3pt]
\es{A María le gusta el café.} & \de{María gefällt Kaffee.} \\[3pt]
\es{No nos gusta cocinar.} & \de{Uns gefällt Kochen nicht.} \\[3pt]
\es{Les gustan los deportes.} & \de{Ihnen gefallen Sportarten.} \\
\end{tabular}}

\vspace{6pt}

\begin{tippbox}
{\footnotesize\textit{Denke immer:} Was ist das SUBJEKT?\\
Singular/Infinitiv $\rightarrow$ \es{gusta}\\
Plural $\rightarrow$ \es{gustan}}
\end{tippbox}
