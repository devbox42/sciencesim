\abschnitt{1}{Verbos en -ar} \de{-- Verben auf -ar}

\vspace{4pt}

\begin{grammatikbox}
\textbf{\gram{Konjugation: hablar}}

\vspace{4pt}
{\small
\begin{tabular}{@{}ll@{}}
yo & habl\stamm{aba} \\[0.3em]
tú & habl\stamm{abas} \\[0.3em]
él/ella/usted & habl\stamm{aba} \\[0.3em]
nosotros/as & habl\stamm{ábamos} \\[0.3em]
vosotros/as & habl\stamm{abais} \\[0.3em]
ellos/ellas/ustedes & habl\stamm{aban} \\
\end{tabular}}
\end{grammatikbox}

\begin{tippbox}
\footnotesize\textit{Merke:} Die 1. und 3. Person Singular sind identisch!
\end{tippbox}

\abschnitt{5}{Imperfecto vs. Indefinido} \de{-- Der wichtigste Unterschied}

\vspace{4pt}

\begin{merkbox}
\textbf{Wann benutze ich welche Zeitform?}

\vspace{4pt}
\begin{multicols}{2}

\textbf{\textcolor{colorrojo}{IMPERFECTO}}\\
Hintergrund, Gewohnheit, Beschreibung

\vspace{4pt}
\begin{tabular}{@{}ll@{}}
$\bullet$ & siempre, normalmente \\
$\bullet$ & cada día, todos los días \\
$\bullet$ & a menudo, frecuentemente \\
$\bullet$ & mientras, cuando (Hintergrund) \\
$\bullet$ & de niño/a, en aquella época \\
\end{tabular}

\columnbreak

\textbf{\textcolor{colorazul}{INDEFINIDO}}\\
Einmalige, abgeschlossene Handlung

\vspace{4pt}
\begin{tabular}{@{}ll@{}}
$\bullet$ & ayer, anteayer \\
$\bullet$ & el año pasado, hace dos días \\
$\bullet$ & de repente, en ese momento \\
$\bullet$ & una vez, dos veces \\
$\bullet$ & en 2020, el lunes pasado \\
\end{tabular}

\end{multicols}
\end{merkbox}

\abmark{G4}

% Eselsbrücke
% ---------------------------------------------------------------
\begin{grammatikbox}
\textbf{Eselsbrücke: Kamera vs. Foto}

\vspace{4pt}
\textbf{Imperfecto} = Die Kamera \textbf{schwenkt} (Beschreibung, Hintergrund, was passierte)

\textbf{Indefinido} = Du \textbf{knipst ein Foto} (einmalige Aktion, abgeschlossenes Ereignis)

\vspace{4pt}
{\small
Beispiel: \es{Mientras \textcolor{colorrojo}{caminaba} por la calle, \textcolor{colorazul}{vi} a María.}\\
\de{Während ich die Straße entlang ging (Hintergrund), sah ich María (einmalig).}}
\end{grammatikbox}
