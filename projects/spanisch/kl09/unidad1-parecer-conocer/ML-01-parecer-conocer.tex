{\small
\textbf{¿Qué te parece?} -- Mit \es{parecer} kannst du deine Meinung sagen.
Mit \es{conocer} fragst du, ob jemand eine Person oder Sache kennt.
Beides brauchst du ständig im Alltag!}

\vspace{8pt}

\begin{grammatikbox}
\begin{multicols}{2}

% --- parecer ---
{\large\bfseries parecer} \de{scheinen, gefallen}

\vspace{4pt}

Konstruktion wie \es{gustar}:

\vspace{3pt}

{\small
\begin{tabular}{@{}ll@{}}
me parece(n) & \de{mir scheint} \\
te parece(n) & \de{dir scheint} \\
le parece(n) & \de{ihm/ihr scheint} \\
\end{tabular}}

\columnbreak

% --- conocer ---
{\large\bfseries conocer} \de{kennen}

\vspace{4pt}

\regel{yo → conozco} \quad {\small(c → zc)}

\vspace{6pt}

{\small
\begin{tabular}{@{}ll@{}}
\es{conocer a} & + Person \\
\es{conocer} & + Sache \\
\end{tabular}}

\end{multicols}

\vspace{4pt}

\begin{center}
{\small
\regel{e→\stamm{ie}} \es{qu\stamm{ie}ro, qu\stamm{ie}res, qu\stamm{ie}re}, queremos, queréis, \es{qu\stamm{ie}ren}
\hspace{6pt}
\regel{o→\stamm{ue}} \es{p\stamm{ue}do, p\stamm{ue}des, p\stamm{ue}de}, podemos, podéis, \es{p\stamm{ue}den}}
\end{center}
\end{grammatikbox}

\begin{multicols}{2}

% --- AUFGABE 1 ---
\aufgabe{1}{Ergänze \es{parecer}.} \punktebox{6}

\begin{teil}
    \item \es{¿Qué te} \lsg{parecen} \es{los deberes?}\\
          --- \es{Me} \lsg{parecen} \es{difíciles.}
    \item \es{¿Qué os} \lsg{parece} \es{la película?}\\
          --- \es{Nos} \lsg{parece} \es{muy divertida.}
    \item \es{A Clara le} \lsg{parece} \es{que Gonzalo es un poco raro.}
    \item \es{¿Qué te} \lsg{parece} \es{quedar a las ocho?}
\end{teil}

\vspace{6pt}

% --- AUFGABE 2 ---
\aufgabe{2}{Ergänze \es{conocer}.} \punktebox{7}

\begin{teil}
    \item \es{Yo no} \lsg{conozco} \es{a tu hermano.}
    \item \es{¿Tú} \lsg{conoces} \es{la calle Trinidad?}
    \item \es{Nosotros} \lsg{conocemos} \es{a toda tu familia.}
    \item \es{Mis hermanos} \lsg{conocen} \es{a Alfonso.}
    \item \es{Clara todavía no} \lsg{conoce} \es{muy bien a Nicolás.}
    \item \es{¿Vosotros} \lsg{conocéis} \es{a los amigos de Gonzalo?}
    \item \es{Yo} \lsg{conozco} \es{todas las calles del centro.}
\end{teil}

\columnbreak

% --- AUFGABE 3 ---
\aufgabe{3}{\es{a} oder keine Präposition?} \punktebox{6}

{\small\textit{Erinnerung:} \es{conocer + a} bei Personen.}

\vspace{4pt}

\begin{teil}
    \item \es{¿Conoces} \lsg{a} \es{mi madre?} {\scriptsize\textcolor{loesung}{(Person)}}
    \item \es{No conozco} \lsg{--} \es{esta ciudad.} {\scriptsize\textcolor{loesung}{(Sache)}}
    \item \es{Queremos conocer} \lsg{a} \es{tus amigos.} {\scriptsize\textcolor{loesung}{(Pers.)}}
    \item \es{¿Conocéis} \lsg{--} \es{el nuevo piso de Clara?} {\scriptsize\textcolor{loesung}{(S.)}}
    \item \es{Todavía no conozco} \lsg{a} \es{Nicolás.} {\scriptsize\textcolor{loesung}{(Person)}}
    \item \es{Mi hermano conoce} \lsg{--} \es{todas las canciones...} {\scriptsize\textcolor{loesung}{(S.)}}
\end{teil}

\end{multicols}

\aufgabe{4}{Eigene Sätze.} \punktebox{7}

\begin{multicols}{2}

\textbf{a)} Beantworte mit \es{parecer}: \hfill{\small(3\,P.)}

\vspace{6pt}

\es{¿Qué te parece tu clase de español?}

\vspace{3pt}
\lsg{Me parece muy interesante. / Me parece divertida.}

\vspace{8pt}

\es{¿Qué te parecen los deberes?}

\vspace{3pt}
\lsg{Me parecen difíciles. / Me parecen fáciles.}

\columnbreak

\textbf{b)} Zwei Sätze mit \es{conocer}: \hfill{\small(4\,P.)}

{\small Achte auf \es{a} bei Personen!}

\vspace{6pt}
Person: \lsg{Conozco a mi profesor de español.}

\vspace{6pt}
Sache: \lsg{Conozco la ciudad de Cádiz.}

\end{multicols}
