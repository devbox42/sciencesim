\textbf{Aufgabe 1: Ergänze die richtige Form von \textit{parecer}.} \punktebox{6}

\begin{enumerate}[label=\alph)}]
    \item ¿Qué te \lsg{parecen} los deberes?
    \item Me \lsg{parecen} muy difíciles.
    \item ¿Qué os \lsg{parece} la película?
    \item Nos \lsg{parece} muy divertida.
    \item A Clara le \lsg{parece} que Gonzalo es raro.
    \item ¿Qué te \lsg{parece} quedar a las ocho?
\end{enumerate}

\textcolor{loesung}{\small Hinweis: parecen bei Plural (los deberes), parece bei Singular (la película, quedar)}

\vspace{0.5cm}

% ============================================================================
% AUFGABE 2
% ============================================================================
\textbf{Aufgabe 2: Ergänze die richtige Form von \textit{conocer}.} \punktebox{7}

\begin{enumerate}[label=\alph)]
    \item Yo no \lsg{conozco} a tu hermano.
    \item ¿Tú \lsg{conoces} la calle Trinidad?
    \item Nosotros \lsg{conocemos} a toda tu familia.
    \item Mis hermanos \lsg{conocen} a Alfonso.
    \item Clara todavía no \lsg{conoce} muy bien a Nicolás.
    \item ¿Vosotros \lsg{conocéis} a los amigos de Gonzalo?
    \item Ado dice: ``Yo \lsg{conozco} todas las calles del centro.''
\end{enumerate}

\textcolor{loesung}{\small Hinweis: yo = conozco (c → zc vor o)}

\vspace{0.5cm}

% ============================================================================
% AUFGABE 3
% ============================================================================
\textbf{Aufgabe 3: Ergänze \textit{a} wo nötig.} \punktebox{6}

\begin{enumerate}[label=\alph)]
    \item ¿Conoces \lsg{a} mi madre? \textcolor{loesung}{\small (Person)}
    \item No conozco \lsg{--} esta ciudad. \textcolor{loesung}{\small (Sache)}
    \item Queremos conocer \lsg{a} tus amigos. \textcolor{loesung}{\small (Personen)}
    \item ¿Conocéis \lsg{--} el nuevo piso de Clara? \textcolor{loesung}{\small (Sache)}
    \item Todavía no conozco \lsg{a} Nicolás. \textcolor{loesung}{\small (Person)}
    \item Mi hermano conoce \lsg{--} todas las canciones. \textcolor{loesung}{\small (Sache)}
\end{enumerate}

\textcolor{loesung}{\small Regel: conocer + a bei PERSONEN!}

\vspace{0.5cm}

\textbf{Aufgabe 4: Kreuze die richtige Antwort an.} \punktebox{6}
