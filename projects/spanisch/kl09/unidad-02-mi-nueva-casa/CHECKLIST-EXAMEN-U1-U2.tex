\tema{Gramática: El pretérito perfecto}

\begin{itemize}[leftmargin=1.2em, itemsep=0pt, parsep=0pt]
    \item[\checkbox] Bildung: \textbf{haber} + \textbf{participio} (he, has, ha, hemos, habéis, han + -ado/-ido)
    \item[\checkbox] Partizip: -ar → -ado / -er → -ido / -ir → -ido
    \item[\checkbox] Unregelmäßig: hecho, dicho, visto, escrito, puesto, vuelto, abierto, muerto
    \item[\checkbox] Signalwörter: hoy, esta semana, ya, todavía no, nunca, alguna vez
\end{itemize}
