{\small Partizipien (participio):}

\vspace{3pt}

{\footnotesize
\begin{tabular}{@{}lll@{}}
\textbf{-ar} & hablar & habl\luecke{0.8cm} \\[0.3em]
\textbf{-er} & comer & com\luecke{0.8cm} \\[0.3em]
\textbf{-ir} & vivir & viv\luecke{0.8cm} \\[0.5em]

\end{tabular}}
