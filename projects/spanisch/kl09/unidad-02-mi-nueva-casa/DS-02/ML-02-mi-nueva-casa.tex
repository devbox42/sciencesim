\begin{grammatikbox}
\sanstext{\footnotesize\textbf{Kasten G1: El pretérito perfecto}}

\vspace{4pt}

\begin{multicols}{2}

{\small Ergänze die Konjugation von \es{haber}:}

\vspace{3pt}

{\footnotesize
\begin{tabular}{@{}ll@{}}
yo & \lsg{he} \\
tú & \lsg{has} \\
él/ella/usted & \lsg{ha} \\
nosotros/-as & \lsg{hemos} \\
vosotros/-as & \lsg{habéis} \\
ellos/ellas/ustedes & \lsg{han} \\
\end{tabular}}

\columnbreak

{\small Partizipien (participio):}

\vspace{3pt}

{\footnotesize
\begin{tabular}{@{}lll@{}}
\textbf{-ar} & hablar & habl\lsg{ado} \\[0.3em]
\textbf{-er} & comer & com\lsg{ido} \\[0.3em]
\textbf{-ir} & vivir & viv\lsg{ido} \\[0.5em]
\multicolumn{3}{@{}l}{\textbf{Unregelmäßig:}} \\[0.3em]
poner & $\rightarrow$ & \lsg{puesto} \\
hacer & $\rightarrow$ & \lsg{hecho} \\
\end{tabular}}

\end{multicols}

\vspace{2pt}

{\footnotesize \regel{Bildung} \es{haber} (konjugiert) + \es{participio} $\rightarrow$ \es{He puesto la lámpara en el salón.}}

\end{grammatikbox}

\vspace{4pt}

\aufgabe{Aufgabe 3}{Lückentext: Der Umzugsbericht.} \punktebox{8}

{\small Ergänze den Text mit dem pretérito perfecto und den Objektpronomen.}

{\small
Esta semana \textbf{(nosotros, mudarse)} \lsg{nos hemos mudado} a un nuevo piso.

El lunes los chicos de la mudanza \textbf{(bajar)} \lsg{han bajado} los muebles del camión.

Mi padre \textbf{(subir)} \lsg{ha subido} las cajas al tercer piso.

Yo \textbf{(poner)} \lsg{he puesto} mi cama en el dormitorio. La lámpara también --

\lsg{la} \textbf{(poner, yo)} \lsg{he puesto} al lado de la cama.

Mi madre \textbf{(sacar)} \lsg{ha sacado} la ropa de las cajas.

¿Y el sofá? Mi hermano \lsg{lo} \textbf{(llevar)} \lsg{ha llevado} al salón.

¡Qué día tan largo!
}

\aufgabe{Aufgabe 4}{Bilde Sätze im pretérito perfecto.} \punktebox{5}

\vspace{2pt}

{\small Schreibe zu jedem Bild einen Satz im pretérito perfecto.}

\vspace{4pt}

\begin{teil}
    \item Clara / poner / la lámpara / en el salón\\
    $\rightarrow$ \lsg{Clara ha puesto la lámpara en el salón.}

    \item Nosotros / subir / los muebles / al piso\\
    $\rightarrow$ \lsg{Nosotros hemos subido los muebles al piso.}

    \item Papá / sacar / la nevera / del camión\\
    $\rightarrow$ \lsg{Papá ha sacado la nevera del camión.}

    \item Yo / llevar / el televisor / al dormitorio\\
    $\rightarrow$ \lsg{Yo he llevado el televisor al dormitorio.}

    \item Mamá / poner / el cuadro / en la pared\\
    $\rightarrow$ \lsg{Mamá ha puesto el cuadro en la pared.}
\end{teil}

\vspace{8pt}
