% ---------------------------------------------------------------
% STATION 2: PRETÉRITO PERFECTO
% ---------------------------------------------------------------
\abschnitt{2}{El pretérito perfecto}

\vspace{4pt}

Wiederhole die Bildung:

\begin{grammatikbox}
\textbf{haber + Partizip}

\vspace{3pt}

{\footnotesize
\begin{tabular}{@{}ll@{}}
yo & \es{he} \\
tú & \es{has} \\
él/ella & \es{ha} \\
nosotros & \es{hemos} \\
vosotros & \es{habéis} \\
ellos & \es{han} \\
\end{tabular}}

\vspace{4pt}

{\footnotesize
\textbf{Partizip:}\\
-ar → \stamm{-ado}\\
-er/-ir → \stamm{-ido}\\
poner → \stamm{puesto}\\
hacer → \stamm{hecho}
}
\end{grammatikbox}

\abmark{Aufgabe 2}
