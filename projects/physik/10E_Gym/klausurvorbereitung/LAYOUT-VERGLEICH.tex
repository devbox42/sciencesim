\vspace{0.5cm}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Gleichförmige Bewegung}

\vspace{0.2cm}
\begin{tcolorbox}[
    colback=physik!5,
    colframe=physik!50,
    boxrule=0.3pt,
    arc=0pt,
    left=3pt, right=3pt, top=3pt, bottom=3pt
]
$s = v \cdot t$\\[0.1cm]
$v = \dfrac{s}{t}$\\[0.1cm]
$t = \dfrac{s}{v}$
\end{tcolorbox}

\vspace{0.3cm}
\textbf{Beschleunigte Bewegung}

\vspace{0.2cm}
\begin{tcolorbox}[
    colback=physik!5,
    colframe=physik!50,
    boxrule=0.3pt,
    arc=0pt,
    left=3pt, right=3pt, top=3pt, bottom=3pt
]
$v = a \cdot t$\\[0.1cm]
$s = \dfrac{1}{2} a \cdot t^2$\\[0.1cm]
$v^2 = 2 \cdot a \cdot s$
\end{tcolorbox}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Freier Fall}

\vspace{0.2cm}
\begin{tcolorbox}[
    colback=physik!5,
    colframe=physik!50,
    boxrule=0.3pt,
    arc=0pt,
    left=3pt, right=3pt, top=3pt, bottom=3pt
]
$g \approx 10\,\mathrm{m/s^2}$\\[0.1cm]
$s = \dfrac{1}{2} g \cdot t^2$\\[0.1cm]
$v = g \cdot t$\\[0.1cm]
$v = \sqrt{2 \cdot g \cdot s}$
\end{tcolorbox}

\vspace{0.3cm}
\textbf{Senkrechter Wurf}

\vspace{0.2cm}
\begin{tcolorbox}[
    colback=physik!5,
    colframe=physik!50,
    boxrule=0.3pt,
    arc=0pt,
    left=3pt, right=3pt, top=3pt, bottom=3pt
]
$t_{\text{steig}} = \dfrac{v_0}{g}$\\[0.2cm]
$h_{\text{max}} = \dfrac{v_0^2}{2g}$
\end{tcolorbox}
\end{minipage}
