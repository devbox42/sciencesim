\aufgabe{Grundlagen}{6}

\begin{enumerate}
\item[a)] \lsg{108 km/h = 108 $\div$ 3,6 = 30 m/s} \be{1}

\item[b)] \lsg{Formel: $s = \frac{1}{2}gt^2$ $\Rightarrow$ $t = \sqrt{\frac{2s}{g}}$} \be{1}\\
\lsg{Einsetzen: $t = \sqrt{\frac{2 \cdot 1{,}25\,\mathrm{m}}{10\,\mathrm{m/s^2}}} = \sqrt{0{,}25\,\mathrm{s^2}} = 0{,}5\,\mathrm{s}$} \be{1}

\item[c)] Tabelle: \be{3}

\begin{center}
\begin{tabular}{|l|c|c|}
\hline
\textbf{Diagramm} & \textbf{Steigung bedeutet} & \textbf{Fläche bedeutet} \\
\hline
$s(t)$-Diagramm & \lsg{Geschwindigkeit $v$} & --- \\
\hline
$v(t)$-Diagramm & \lsg{Beschleunigung $a$} & \lsg{Weg $s$} \\
\hline
\end{tabular}
\end{center}
{\small\color{correct} Je 1 BE pro richtige Eintragung}
\end{enumerate}

\aufgabe{Gefahrenbremsung im Straßenverkehr}{8}

\begin{enumerate}
\item[a)] \lsg{$v_0 = 72\,\mathrm{km/h} = 72 \div 3{,}6 = 20\,\mathrm{m/s}$} \be{1}

\item[b)] \lsg{Formel: $v = v_0 + a \cdot t$ $\Rightarrow$ $t = \dfrac{v_0}{|a|}$} \be{1}\\[0.2cm]
\lsg{Einsetzen: $t = \dfrac{20\,\mathrm{m/s}}{5\,\mathrm{m/s^2}} = 4\,\mathrm{s}$} \be{1}

\item[c)] \lsg{Formel: $s = \dfrac{v_0^2}{2|a|}$} \be{1}\\[0.2cm]
\lsg{Einsetzen: $s = \dfrac{(20\,\mathrm{m/s})^2}{2 \cdot 5\,\mathrm{m/s^2}} = \dfrac{400}{10}\,\mathrm{m} = 40\,\mathrm{m}$} \be{1}

\item[d)] \lsg{Nein, der Fahrer kann nicht rechtzeitig anhalten.} \be{1}\\
\lsg{Begründung: Der Bremsweg (40 m) ist länger als der Abstand zum Kind (35 m).} \be{1}\\
\lsg{Das Kind wird 5 m vor dem Stillstand erreicht: $40\,\mathrm{m} - 35\,\mathrm{m} = 5\,\mathrm{m}$ fehlen.} \be{1}
\end{enumerate}

\aufgabe{Basketball-Wurf}{8}

\begin{enumerate}
\item[a)] \lsg{Formel: $v = v_0 - g \cdot t$ $\Rightarrow$ $t_{\text{steig}} = \dfrac{v_0}{g}$} \be{1}\\[0.2cm]
\lsg{Einsetzen: $t_{\text{steig}} = \dfrac{15\,\mathrm{m/s}}{10\,\mathrm{m/s^2}} = 1{,}5\,\mathrm{s}$} \be{1}

\item[b)] \lsg{Formel: $h_{\text{max}} = \dfrac{v_0^2}{2g}$} \be{1}\\[0.2cm]
\lsg{Einsetzen: $h_{\text{max}} = \dfrac{(15\,\mathrm{m/s})^2}{2 \cdot 10\,\mathrm{m/s^2}} = \dfrac{225}{20}\,\mathrm{m} = 11{,}25\,\mathrm{m}$} \be{2}

\item[c)] \lsg{Der Ball kommt mit $v = 15\,\mathrm{m/s}$ (nach unten) an.} \be{1}\\
\lsg{Begründung: Energieerhaltung / Symmetrie der Bewegung / zeitlose Formel liefert $|v| = v_0$} \be{2}

{\small\color{correct} 2 BE für korrekte Begründung, 1 BE für richtige Geschwindigkeit}
\end{enumerate}

\aufgabe{E-Scooter-Fahrt}{8}

\begin{enumerate}
\item[a)] \lsg{Phase A (0--4 s): Beschleunigte Bewegung, der E-Scooter beschleunigt von 0 auf 8 m/s.} \be{1}\\
\lsg{Phase B (4--8 s): Gleichförmige Bewegung mit konstant 8 m/s.} \be{1}\\
\lsg{Phase C (8--12 s): Bremsvorgang, der E-Scooter bremst von 8 m/s auf 0 ab.} \be{1}

\item[b)] \lsg{Formel: $a = \dfrac{\Delta v}{\Delta t}$} \be{1}\\[0.2cm]
\lsg{Einsetzen: $a = \dfrac{8\,\mathrm{m/s} - 0}{4\,\mathrm{s}} = 2\,\mathrm{m/s^2}$} \be{1}

\item[c)] \lsg{Fläche unter dem Graphen = Weg. Zerlegung in geometrische Figuren:} \be{1}\\
\lsg{Phase A (Dreieck): $s_A = \dfrac{1}{2} \cdot 4\,\mathrm{s} \cdot 8\,\mathrm{m/s} = 16\,\mathrm{m}$}\\
\lsg{Phase B (Rechteck): $s_B = 4\,\mathrm{s} \cdot 8\,\mathrm{m/s} = 32\,\mathrm{m}$}\\
\lsg{Phase C (Dreieck): $s_C = \dfrac{1}{2} \cdot 4\,\mathrm{s} \cdot 8\,\mathrm{m/s} = 16\,\mathrm{m}$}\\
\lsg{Gesamtweg: $s = 16 + 32 + 16 = 64\,\mathrm{m}$} \be{2}

{\small\color{correct} Alternativ: Trapez-Formel für A+B+C zusammen}
\end{enumerate}

\aufgabe{Fallschirmspringen}{8}

\begin{enumerate}
\item[a)] \lsg{Beim Öffnen des Fallschirms vergrößert sich die Querschnittsfläche $A$ stark.} \be{1}\\
\lsg{Da $F_W \sim A$ (und $F_W \sim v^2$), wird der Luftwiderstand viel größer.} \be{1}\\
\lsg{Der Fallschirmspringer erreicht eine niedrigere Grenzgeschwindigkeit.} \be{1}

{\small\color{correct} Relevante Größen z.\,B.: Fläche $A$, Luftwiderstandskraft $F_W$, Grenzgeschwindigkeit $v_{\text{grenz}}$}

\item[b)] \lsg{In 39 km Höhe ist die Luftdichte $\rho$ viel geringer als am Boden.} \be{1}\\
\lsg{Weniger Luftdichte bedeutet weniger Luftwiderstand.} \be{1}\\
\lsg{Dadurch ist die Grenzgeschwindigkeit höher.} \be{1}

\item[c)] \lsg{Die Aussage ist \textbf{falsch}. Das Experiment auf dem Mond beweist es.} \be{1}\\
\lsg{Begründung: Im Vakuum gibt es keinen Luftwiderstand. Alle Körper fallen gleich schnell, unabhängig von ihrer Masse. Die Formel $s = \frac{1}{2}gt^2$ enthält keine Masse $m$.} \be{1}
\end{enumerate}

\aufgabe{Vergleich zweier Bewegungen}{10}

\begin{enumerate}
\item[a)] \lsg{Aus dem $s(t)$-Diagramm: $v_A = \dfrac{\Delta s}{\Delta t} = \dfrac{100\,\mathrm{m}}{10\,\mathrm{s}} = 10\,\mathrm{m/s}$} \be{2}

\item[b)] \lsg{Aus dem $v(t)$-Diagramm: $a_B = \dfrac{\Delta v}{\Delta t} = \dfrac{20\,\mathrm{m/s}}{10\,\mathrm{s}} = 2\,\mathrm{m/s^2}$} \be{2}

\item[c)] \lsg{Fläche unter dem $v(t)$-Diagramm = Dreieck}\\
\lsg{$s_B = \dfrac{1}{2} \cdot 10\,\mathrm{s} \cdot 20\,\mathrm{m/s} = 100\,\mathrm{m}$} \be{2}

\item[d)] \lsg{Beide Autos legen 100 m zurück, also gleich weit.}\\
\lsg{Auto A: direkt ablesbar (100 m). Auto B: berechnet (100 m).} \be{2}

\item[e)] \lsg{Auto A hat konstant $v_A = 10\,\mathrm{m/s}$.}\\
\lsg{Auto B erreicht diese Geschwindigkeit bei $t = \dfrac{v}{a} = \dfrac{10}{2} = 5\,\mathrm{s}$} \be{2}
\end{enumerate}

\aufgabe{Reaktionsweg und Anhalteweg}{10}

\begin{enumerate}
\item[a)] \lsg{Formel: $s_{\text{Reaktion}} = v \cdot t_R$}\\
\lsg{Einsetzen: $s_{\text{Reaktion}} = 15\,\mathrm{m/s} \cdot 1\,\mathrm{s} = 15\,\mathrm{m}$} \be{2}

\item[b)] \lsg{Formel: $s_{\text{Brems}} = \dfrac{v^2}{2|a|}$}\\
\lsg{Einsetzen: $s_{\text{Brems}} = \dfrac{(15\,\mathrm{m/s})^2}{2 \cdot 5\,\mathrm{m/s^2}} = \dfrac{225}{10}\,\mathrm{m} = 22{,}5\,\mathrm{m}$} \be{2}

\item[c)] \lsg{$s_{\text{Anhalt}} = 15\,\mathrm{m} + 22{,}5\,\mathrm{m} = 37{,}5\,\mathrm{m}$} \be{1}

\item[d)] \lsg{Bei $v = 30\,\mathrm{m/s}$:}\\
\lsg{$s_{\text{Reaktion}} = 30\,\mathrm{m/s} \cdot 1\,\mathrm{s} = 30\,\mathrm{m}$} \be{1}\\
\lsg{$s_{\text{Brems}} = \dfrac{(30\,\mathrm{m/s})^2}{10\,\mathrm{m/s^2}} = \dfrac{900}{10}\,\mathrm{m} = 90\,\mathrm{m}$} \be{1}\\
\lsg{$s_{\text{Anhalt}} = 30\,\mathrm{m} + 90\,\mathrm{m} = 120\,\mathrm{m}$} \be{1}

\item[e)] \lsg{Der Bremsweg hängt quadratisch von der Geschwindigkeit ab ($s \sim v^2$).}\\
\lsg{Doppelte Geschwindigkeit bedeutet vierfachen Bremsweg.}\\
\lsg{Der Reaktionsweg ist nur linear (doppelt). Zusammen ergibt das mehr als das Doppelte.} \be{2}
\end{enumerate}

\lsg{Ansatz: Positionen beider Autos als Funktion der Zeit aufstellen.}\\[0.2cm]

\lsg{Gleichsetzen: $t^2 = 20t - 100$}\\
\lsg{$t^2 - 20t + 100 = 0$}\\
\lsg{$(t - 10)^2 = 0$}\\
\lsg{$t = 10\,\mathrm{s}$} \be{2}

{\small\color{correct} 1 BE für korrekten Ansatz, 1 BE für richtige Lösung. Probe: $s_1(10) = 100\,\mathrm{m}$, $s_2(10) = 100\,\mathrm{m}$ \checkmark}
