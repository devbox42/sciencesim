\sectionbox{Gleichförmige Bewegung}{
\small
$v = \dfrac{s}{t}$ \quad $s = v \cdot t$ \quad $t = \dfrac{s}{v}$

\vspace{0.1cm}
Mit Startposition: $s(t) = v \cdot t + s_0$
}

\sectionbox{Beschleunigte Bewegung}{
\small
$a = \dfrac{\Delta v}{\Delta t} = \dfrac{v - v_0}{t}$

\vspace{0.15cm}
$v(t) = a \cdot t + v_0$

$s(t) = \frac{1}{2} a \cdot t^2 + v_0 \cdot t + s_0$

\vspace{0.15cm}
\textbf{Zeitlos:} $v^2 = v_0^2 + 2 \cdot a \cdot s$

\vspace{0.15cm}
\textbf{Aus Ruhe:} $v = at$ \quad $s = \frac{1}{2}at^2$
}

\sectionbox{Bremsvorgang}{
\small
Verzögerung: $a < 0$

$t_{\text{brems}} = \dfrac{v_0}{|a|}$ \qquad $s_{\text{brems}} = \dfrac{v_0^2}{2|a|}$

\vspace{0.1cm}
\textit{2× Tempo = 4× Bremsweg}
}

\sectionbox{Freier Fall}{
\small
$g \approx 10\,\mathrm{m/s^2}$

\vspace{0.1cm}
\begin{tabular}{@{}ll@{}}
Geschwindigkeit: & $v = g \cdot t = 10t$ \\
Fallstrecke: & $s = \frac{1}{2}gt^2 = 5t^2$ \\
Fallzeit: & $t = \sqrt{s/5}$ \\
Aufprall: & $v = \sqrt{2gs}$
\end{tabular}
}

\sectionbox{Senkrechter Wurf nach oben}{
\small
$v(t) = v_0 - g \cdot t$ \quad (bremst)

Steigzeit: $t_{\text{steig}} = \dfrac{v_0}{g}$

Steighöhe: $h_{\text{max}} = \dfrac{v_0^2}{2g}$
}
