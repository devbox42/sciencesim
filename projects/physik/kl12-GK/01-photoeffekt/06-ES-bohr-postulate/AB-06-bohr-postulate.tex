\begin{tcolorbox}[infobox]
\textbf{Vorschau: Bohrs Postulate (1913)}

\vspace{0.3em}

\begin{enumerate}[noitemsep,topsep=0pt]
\item Elektronen auf bestimmten Bahnen strahlen \textbf{nicht}.
\item Nur Bahnen mit $n = 1, 2, 3, ...$ sind erlaubt (\textbf{Quantisierung}).
\item Bei Übergängen: $E_{\text{Photon}} = |E_{n_1} - E_{n_2}|$
\end{enumerate}

\vspace{0.3em}

\textbf{Am Freitag:} Rechnen mit $E_n = \dfrac{-13{,}6 \text{ eV}}{n^2}$ und Rydberg-Formel
\end{tcolorbox}
