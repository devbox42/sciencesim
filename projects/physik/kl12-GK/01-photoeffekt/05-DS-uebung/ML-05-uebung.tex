{\large\textbf{Teil A: Photoeffekt}} \hfill \textbf{(12 BE)}

\vspace{0.3em}

\textbf{Aufgabe 1:} Eine Kaliumkathode ($W_A = 2{,}3$ eV) wird mit Licht der Wellenlänge $\lambda = 400$ nm bestrahlt.

\begin{enumerate}[label=\alph*)}
\item Berechne die Energie eines Photons in eV. \hfill (3 BE)

\lsg{$f = \dfrac{c}{\lambda} = \dfrac{3 \cdot 10^8 \text{ m/s}}{400 \cdot 10^{-9} \text{ m}} = 7{,}5 \cdot 10^{14}$ Hz}

\lsg{$E = h \cdot f = 4{,}14 \cdot 10^{-15} \text{ eV}\cdot\text{s} \cdot 7{,}5 \cdot 10^{14} \text{ Hz} = 3{,}1$ eV}

\vspace{0.3em}

\item Berechne die maximale kinetische Energie der ausgelösten Elektronen. \hfill (2 BE)

\lsg{$E_{\text{kin,max}} = h \cdot f - W_A = 3{,}1 \text{ eV} - 2{,}3 \text{ eV} = 0{,}8$ eV}

\vspace{0.3em}

\item Bestimme die Grenzwellenlänge für Kalium. \hfill (3 BE)

\lsg{Bei der Grenzfrequenz gilt: $h \cdot f_G = W_A$}

\lsg{$f_G = \dfrac{W_A}{h} = \dfrac{2{,}3 \text{ eV}}{4{,}14 \cdot 10^{-15} \text{ eV}\cdot\text{s}} = 5{,}56 \cdot 10^{14}$ Hz}

\lsg{$\lambda_G = \dfrac{c}{f_G} = \dfrac{3 \cdot 10^8 \text{ m/s}}{5{,}56 \cdot 10^{14} \text{ Hz}} = 540$ nm (grün)}

\vspace{0.3em}

\end{enumerate}

{\large\textbf{Teil B: De-Broglie-Wellenlänge}} \hfill \textbf{(10 BE)}

\vspace{0.3em}

\textbf{Aufgabe 3:} Elektronen werden mit einer Spannung $U_B = 5{,}0$ kV beschleunigt.

\begin{enumerate}[label=\alph*)]
\item Berechne die kinetische Energie der Elektronen in Joule. \hfill (2 BE)

\lsg{$E_{\text{kin}} = e \cdot U_B = 1{,}60 \cdot 10^{-19} \text{ C} \cdot 5000 \text{ V} = 8{,}0 \cdot 10^{-16}$ J}

\vspace{0.3em}

\item Berechne den Impuls der Elektronen. \hfill (3 BE)

\textit{Hinweis:} $E_{\text{kin}} = \dfrac{p^2}{2m}$ $\Rightarrow$ $p = \sqrt{2 \cdot m_e \cdot E_{\text{kin}}}$

\lsg{$p = \sqrt{2 \cdot 9{,}11 \cdot 10^{-31} \text{ kg} \cdot 8{,}0 \cdot 10^{-16} \text{ J}}$}

\lsg{$p = \sqrt{1{,}46 \cdot 10^{-45}} = 3{,}82 \cdot 10^{-23}$ kg$\cdot$m/s}

\vspace{0.3em}

\item Berechne die De-Broglie-Wellenlänge der Elektronen in pm. \hfill (2 BE)

\lsg{$\lambda = \dfrac{h}{p} = \dfrac{6{,}63 \cdot 10^{-34} \text{ J}\cdot\text{s}}{3{,}82 \cdot 10^{-23} \text{ kg}\cdot\text{m/s}} = 1{,}74 \cdot 10^{-11}$ m $= 17{,}4$ pm}

\vspace{0.3em}

\end{enumerate}

\textbf{Aufgabe 4:} Ein Elektron hat die De-Broglie-Wellenlänge $\lambda = 100$ pm.

\begin{enumerate}[label=\alph*), start=4]
\item Berechne den Impuls des Elektrons. \hfill (2 BE)

\lsg{$p = \dfrac{h}{\lambda} = \dfrac{6{,}63 \cdot 10^{-34} \text{ J}\cdot\text{s}}{100 \cdot 10^{-12} \text{ m}} = 6{,}63 \cdot 10^{-24}$ kg$\cdot$m/s}

\vspace{0.3em}

\item Vergleiche: Welche Wellenlänge hat ein Photon mit gleichem Impuls? \hfill (1 BE)

\lsg{Für ein Photon gilt ebenfalls $p = h/\lambda$, also $\lambda_{\text{Photon}} = 100$ pm (identisch!)}

\lsg{Dies folgt aus der Äquivalenz der De-Broglie-Beziehung für Materie und Photonen.}

\vspace{0.3em}

\end{enumerate}

{\large\textbf{Teil C: Doppelspalt und Komplementarität}} \hfill \textbf{(8 BE)}

\vspace{0.3em}

\textbf{Aufgabe 5:} Elektronen werden einzeln durch einen Doppelspalt geschickt.

\begin{enumerate}[label=\alph*)]
\item Beschreibe, was man auf dem Schirm beobachtet, wenn nur wenige Elektronen auftreffen. \hfill (2 BE)

\lsg{Man sieht einzelne Punkte an scheinbar zufälligen Positionen. Jedes Elektron landet als Punkt auf dem Schirm (Teilchenverhalten beim Nachweis).}

\vspace{0.3em}

\item Beschreibe, was man beobachtet, wenn sehr viele Elektronen aufgetroffen sind. \hfill (2 BE)

\lsg{Die Punkte bilden ein Interferenzmuster mit hellen und dunklen Streifen. An manchen Stellen treffen viele Elektronen auf (Maxima), an anderen fast keine (Minima).}

\vspace{0.3em}

\item Erkläre, warum dieses Verhalten mit dem klassischen Teilchenbild nicht vereinbar ist. \hfill (2 BE)

\lsg{Klassische Teilchen würden nur zwei Streifen hinter den Spalten erzeugen. Das Interferenzmuster zeigt, dass jedes Elektron als Welle durch beide Spalte geht und mit sich selbst interferiert.}

\vspace{0.3em}

\item Was passiert mit dem Muster, wenn man einen Detektor aufstellt, der misst, durch welchen Spalt jedes Elektron geht? Begründe mit dem Komplementaritätsprinzip. \hfill (2 BE)

\lsg{Das Interferenzmuster verschwindet! Nach dem Komplementaritätsprinzip (Bohr) kann man entweder den Weg messen (Teilchenbild) oder Interferenz beobachten (Wellenbild) -- aber nicht beides gleichzeitig.}

\end{enumerate}
