\begin{loesungskasten}
\textbf{Rechenbeispiel Kühlschrank:}
\begin{itemize}[nosep, leftmargin=1.5em]
    \item $E_{\text{Tag}} = P \cdot t = 15\,\text{W} \cdot 24\,\text{h} = 360\,\text{Wh}$
    \item $E_{\text{Jahr}} = 360\,\text{Wh} \cdot 365 \div 1000 = 131{,}4\,\text{kWh}$
    \item Kosten $= 131{,}4\,\text{kWh} \cdot 0{,}30\,\text{€} = 39{,}42\,\text{€}$
\end{itemize}
\end{loesungskasten}

\aufgabe{2}{– Stromkosten berechnen} \punktebox{7}

\begin{teil}
    \item \operator{Berechne} den Tagesverbrauch (Summe aller Geräte). \hfill (2 BE)

    $E_{\text{Tag}} = $ \lsg{6535 Wh} = \lsg{6,54 kWh}

    \item \operator{Berechne} die monatlichen Stromkosten (30 Tage, 30 ct/kWh). \hfill (3 BE)

    \lsg{$E_{\text{Monat}} = 6{,}54\,\text{kWh} \cdot 30 = 196{,}2\,\text{kWh}$}

    \lsg{Kosten $= 196{,}2\,\text{kWh} \cdot 0{,}30\,\text{€/kWh} = 58{,}86\,\text{€}$}

    Kosten pro Monat: \lsg{58,86 €}

    \item \operator{Berechne} die Jahreskosten (365 Tage). \hfill (2 BE)

    \lsg{Kosten $= 6{,}54\,\text{kWh} \cdot 365 \cdot 0{,}30\,\text{€} \approx 716\,\text{€}$}

    (Oder aus Tabelle: Summe der Jahreskosten $\approx$ \lsg{395 €} bei reduzierter Nutzung)
\end{teil}

\aufgabe{4}{– Einsparpotenzial berechnen} \punktebox{4}

Gewählte Maßnahme: \lsg{Wäsche an der Luft trocknen statt Trockner}

\vspace{0.3em}
\begin{tabular}{@{}p{7.5cm}p{7.5cm}@{}}
\textbf{Vorher:} & \textbf{Nachher:} \\
$P$: \lsg{2500} W \quad $t$: \lsg{1} h/Tag & $P$: \lsg{0} W \quad $t$: \lsg{0} h/Tag \\
$E$: \lsg{2500} Wh/Tag & $E$: \lsg{0} Wh/Tag
\end{tabular}

\vspace{0.3em}
\textbf{Ersparnis:} $\Delta E_{\text{Tag}} = $ \lsg{2500} Wh/Tag

Bei 52 Nutzungstagen/Jahr: $\Delta E_{\text{Jahr}} = 2500 \cdot 52 \div 1000 = $ \lsg{130 kWh/Jahr}

\vspace{0.2em}
\textbf{Geldersparnis:} $130\,\text{kWh} \cdot 0{,}30\,\text{€/kWh} = $ \lsg{39,00 €/Jahr}

\vspace{0.3em}
\begin{formelkasten}
\centering
\textbf{Unser Spar-Tipp:} Mit \textit{``\lsg{Lufttrocknen statt Trockner}''} sparen wir ca. \lsg{39 €/Jahr}!
\end{formelkasten}

\vspace{0.5em}
\begin{loesungskasten}
\textbf{Weitere gute Sparmaßnahmen:}
\begin{itemize}[nosep, leftmargin=1.5em]
    \item LED statt Glühbirne: spart ca. 80\% (bei 10 Lampen: $\approx$ 50 €/Jahr)
    \item Standby komplett aus (Steckerleiste): spart ca. 10--15 €/Jahr
    \item Kühlschrank auf 7°C statt 5°C: spart ca. 12\% ($\approx$ 5 €/Jahr)
    \item Waschen mit 30°C statt 60°C: spart ca. 50\% ($\approx$ 10 €/Jahr)
\end{itemize}
\end{loesungskasten}
